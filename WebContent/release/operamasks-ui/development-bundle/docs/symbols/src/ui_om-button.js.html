 14 ;(function($) {
 15 	/**
 16      * @name omButton
 17      * @author 罗业刚
 18      * @class 按钮组件。类似于html中的button、input[type=submit]、input[type=button]，使用背景图片实现圆角，支持icon，可以只显示icon。<br/><br/>
 19      * <b>特点：</b><br/>
 20      * <ol>
 21      *      <li>实现圆角</li>
 22      *      <li>支持左icon和右icon，可同时出现左右icon，也可以只显示icon不显示label</li>
 23      *      <li>按钮文字数目不限，也可以任意设置按钮宽度</li>
 24      *      <li>支持input[type=button]、input[type=submit]、input[type=reset]、button、a五种标签形式</li>
 25      * </ol>
 26      * 
 27      * <b>示例：</b><br/>
 28      * <pre>
 29      * <script type="text/javascript" >
 30      * $(document).ready(function() {
 31      *     $('#bnt').omButton({
 32      *         icons : {left:'images/help.png',right:'images/edit_add.png'},
 33      *         width : 150,
 34      *         disabled : 'disabled',
 35      *         onClick : function(event){
 36      *             // do something
 37      *         }
 38      *     });
 39      * });
 40      * </script>
 41      * 
 42      * <input id="btn" type="submit" />
 43      * </pre>
 44      * 
 45      * @constructor
 46      * @description 构造函数. 
 47      * @param p 标准config对象：{}
 48      */
 49     $.widget('om.omButton', {
 50     	
 51         options: /**@lends omButton# */{
 52         	/**
 53         	 * 是否禁用组件。可通过('#id').attr('disabled')判断按钮是否禁用。
 54              * @type String 
 55              * @default null 
 56              * @example
 57              * $("#button").omButton({disabled: true}); 
 58              */
 59             disabled : null ,
 60             /**
 61              * 显示文本。label值可以写到dom元素里面，也可以设置在属性里面，设置到属性里面的优先级最高。
 62              * @type String 
 63              * @default ""
 64              * @example
 65              * $("#button").omButton({label: "apusic"});
 66              */
 67             label : null ,
 68             /**
 69         	 * 显示按钮图标，left表示左图标，right表示右图标，取值均为图片路径。
 70              * @type Object 
 71              * @default null 
 72              * @example
 73              * $("#button").omButton({
 74              *     icons: {
 75              *         left: 'images/help.png',
 76              *         right: 'images/edit_add.png'
 77              *     }
 78              * });
 79              */
 80             icons: {
 81     			left: null,
 82     			right: null
 83     		},
 84     		/**
 85         	 * 按钮宽度，设置固定宽度之后按钮将不会随文字的多少而改变。
 86              * @type Number 
 87              * @default null 
 88              * @example
 89              * width : 150
 90              */
 91             width : null ,
 92             /**
 93         	 * 单击按钮触发事件。
 94              * @event
 95              * @name omButton#onClick 
 96              * @example
 97              * onClick : function(){
 98              *    //do something
 99              * }
100              */
101             onClick : null
102         },
103         _create:function(){
104             var self = this;
105             var disabledState = this.options.disabled;
106             if ( typeof disabledState != "boolean" ) {
107                 disabledState = this.element.propAttr( "disabled" );
108     		}
109             this._determineButtonType();
110             var options = this.options;
111             if ( this.buttonElement.attr('disabled') == 'disabled' || options.disabled == 'disabled') {
112     			options.disabled = true;
113     		}
114             this._initButton();
115             if(options.disabled){
116             	self._addClass('disabled');
117             	this.buttonElement.css('cursor','default');
118             }
119             var newButtonElement = this.buttonElement.parent().parent();
120             newButtonElement.bind( 'mouseenter.button',function( event ){
121             					if ( options.disabled ) {
122             						return false;
123             					}
124             					self._addClass('hover');
125             				}).bind( "mouseleave.button", function( event ) {
126             					if ( options.disabled ) {
127             						return false;
128             					}
129             					self._removeClass('hover');
130             					self._removeClass('active');
131             				}).bind( "click.button", function( event ){
132             					if ( options.disabled ) {
133             						event.preventDefault();
134             						event.stopImmediatePropagation();
135             						return false;
136             					}
137             				}).bind( "mousedown.button", function( event ) {
138             					if ( options.disabled ) {
139             						return false;
140             					}
141             					self._addClass('active');
142             				    self._removeClass('focus');
143             					var onClick = options.onClick;
144         		                if ( onClick && onClick( event ) === false ) {
145         		                    return;
146         		                }
147             				})
148             				.bind( "mouseup.button", function( event ) {
149             					if ( options.disabled ) {
150             						return false;
151             					}
152             					self._addClass('focus');
153             					self._removeClass('active');
154             				})
155             				.bind( "keydown.button", function(event) {
156             					if ( options.disabled ) {
157             						return false;
158             					}
159             					if ( event.keyCode == $.ui.keyCode.SPACE || event.keyCode == $.ui.keyCode.ENTER ) {
160             						self._addClass('active');
161             					}
162             					if( event.keyCode == $.ui.keyCode.SPACE){
163             						var onClick = options.onClick;
164             		                if ( onClick && onClick( event ) === false ) {
165             		                    return;
166             		                }
167             					}
168             				})
169             				.bind( "keyup.button", function() {
170             					self._removeClass('active');
171             				});
172 	            this.element.bind( "focus", function( event ) {
173 								if ( options.disabled ) {
174 									return false;
175 								}
176 								self._addClass('focus');
177 							}).bind( "blur", function( event ) {
178 	        					if ( options.disabled ) {
179 	        						return false;
180 	        					}
181 	        					self._removeClass('focus');
182 	        				});
183         },
184         /**
185          * 启用组件。
186          * @name omButton#enable
187          * @function
188          * @example
189          * $('#btn').omButton('enable');
190          */
191         enable : function(){
192             this._removeClass('disabled');
193             this.options.disabled = false;
194             this.buttonElement.css('cursor','pointer');
195             return this.element.removeAttr('disabled');
196         },
197         /**
198          * 禁用组件。
199          * @name omButton#disable
200          * @function
201          * @example
202          * $('#btn').omButton('disable');
203          */
204         disable : function(){
205         	this._addClass('disabled');
206             this.options.disabled = true;
207             this.buttonElement.css('cursor','default');
208             return this.element.attr('disabled', 'disabled');
209         },
210         /**
211          * 触发点击事件。
212          * @name omButton#click
213          * @function
214          * @example
215          * $('#btn').omButton('click');
216          */
217         click : function(){
218         	if(!this.options.disabled && this.options.onClick){
219 	        		this.options.onClick();
220             }
221         },
222         /**
223          * 设置按钮的label属性。
224          * @name omButton#changeLabel
225          * @function
226          * @param label 按钮文本
227          * @example
228          * $('#btn').omButton('changeLabel','按钮label');
229          */
230         changeLabel : function(label){
231             if(this.type == 'a'){
232             	this.buttonElement.text(label) ;
233             }else if( this.type == 'input' ){
234             	this.buttonElement.val(label) ;
235             }else if ( this.type == 'button' ){
236             	this.buttonElement.html(label) ;
237             }
238         },
239         /**
240          * 设置按钮的icon属性。
241          * @name omButton#changeIcons
242          * @function
243          * @param icons 图标路径
244          * @example
245          * $('#btn').omButton('changeIcons',{
246          *     left: 'images/help.png',
247          *     right: 'images/edit_add.png'
248          * });
249          */
250         changeIcons : function( icons ){
251         	if( !this.options.disabled ){
252 	            if( icons ){
253 	            	icons.left?this.buttonElement.css( 'backgroundImage','url( '+icons.left+' )' ):null;
254 	            	icons.right?this.buttonElement.next().attr( 'src',icons.right ):null;
255 	            }
256             }
257         },
258         _addClass : function( state ){
259         	this.buttonElement.parent().parent().addClass( 'om-state-'+state );
260         },
261         _removeClass : function( state ){
262         	this.buttonElement.parent().parent().removeClass( 'om-state-'+state );
263         },
264         _initButton : function(){
265         	var options = this.options;
266         	var label = this._getLabel();
267         	var wrapperSpan = $( '<span>' ).addClass( 'om-btn om-state-default' ),
268         	    leftSpan = $( '<span>' ).addClass( 'om-btn-bg om-btn-left' ),
269         	    centerSpan = $( '<span>' ).addClass( 'om-btn-bg om-btn-center' ),
270         	    rightSpan = $( '<span>' ).addClass( 'om-btn-bg om-btn-right' );
271         	if( options.width > 10 ){
272         		centerSpan.css( 'width',parseInt( options.width )-10 );
273         	}
274         	var buttonElement = this.buttonElement.addClass( 'om-btn-txt' );
275         	if( this.type == 'a' || this.type == 'button' ){
276         	    buttonElement.html( label );
277         	}else{
278         	    buttonElement.val( label );
279         	}
280         	if( options.icons.left ){
281         	    if( label ){
282         	        buttonElement.addClass( 'om-btn-icon' ).css( 'background-image','url('+options.icons.left+')' );
283         	    }else{
284         	        buttonElement.addClass( 'om-btn-only-icon' ).css('background-image','url('+options.icons.left+')' );
285         	    }
286         	}
287         	buttonElement.css( {'background-position':'left center','background-repeat':'no-repeat'} );
288         	buttonElement.wrap( wrapperSpan );
289         	leftSpan.insertBefore( buttonElement );
290         	rightSpan.insertAfter( buttonElement );
291         	buttonElement.wrap( centerSpan );
292         	if( options.icons.right ){
293         	    if( label != '' ){
294         	        $( '<img>' ).attr( 'src',options.icons.right ).css( {'vertical-align':'text-top','padding-left':'3px'} ).insertAfter( buttonElement );
295         	    }else{
296         	        $( '<img>' ).attr( 'src',options.icons.right ).css( 'vertical-align','text-top' ).insertAfter( buttonElement );
297         	    }
298             }
299         },
300         _getLabel : function(){
301         	return this.options.label || this.buttonElement.html() || this.buttonElement.text() || this.buttonElement.val();
302         },
303         _determineButtonType: function() {
304     		if ( this.element.is("input") ) {
305     			this.type = "input";
306     		}  else if ( this.element.is("a") ) {
307     			this.type = "a";
308     		} else if ( this.element.is('button') ){
309     			this.type = "button";
310     		}
311     		this.buttonElement = this.element;
312     	}
313     });
314 })(jQuery);
