453     /**
454      * 修正索引，返回值为-1,0,1,2,...,len-1
455      * 只有组件第一次创建的时候才有可能返回-1
456      */
457     _correctIndex: function(index){
458     	var acc = this.element,
459         	options = this.options,
460         	panels = $.data(this.element , "panels"),
461         	panel = acc.children().find(">div#"+index),
462         	len = panels.length,
463         	oldIndex = index;
464         index = panel.length ? $(panels).index(panel) : index;
465         index = index==-1 ? oldIndex : index;
466         //如果id找不到，则认为是索引，进行修正
467         var r = parseInt(index);
468         r = (isNaN(r) && '0' || r)-0;
469         return len==0 || r==-1&&!acc.hasCreated? -1: (r<0? 0 : (r>=len?len-1 : r));
470     },
471     _renderPanels: function () {
472         var acc = this.element,
473         	self = this,
474             panels = [],
475             options = this.options;
476         var headers = acc.find("ul:first");
477         headers.find("a").each(function(n){
478             var href  = this.getAttribute('href', 2);
479             var hrefBase = href.split( "#" )[ 0 ],
480                 baseEl;
481             if ( hrefBase && ( hrefBase === location.toString().split( "#" )[ 0 ] ||
482                     ( baseEl = $( "base" )[ 0 ]) && hrefBase === baseEl.href ) ) {
483                 href = this.hash;
484                 this.href = href;
485             }
486             var anchor = $(this);
487             var cfg = {
488                     title : anchor.text(),
489                     tools : ['collapse'],
490                     iconCls: anchor.attr("iconCls"),
491                     clickExpand : true,
492                     onExpand : function() {
493                         options.onActivate(n);
494                     },
495                     collapsed : true,
496                     onCollapse : function() {
497                         if(acc.hasCreated){
498                             options.onCollapse(n);
499                         }
500                     },
501                     border : true
502             };
503             var target = $('>' + href, acc);
504             var pId = target.prop("id"); 
505             //target要么指向一个当前页面的div，要么是一个url
506             if (!!(target[0])) {
507                 if(!pId){
508                     target.prop("id" , panelIdPrefix+(id++));
509                 }
510                 cfg.content = target.html();
511                 panels.push(self._createPanel(target[0], cfg));
512             } else {
513                 cfg.url = anchor.attr('href');
514                 var panel = self._createPanel($('<div></div>')[0], cfg);
515                 panels.push(panel);
516                 var aid = anchor.prop('id');
517                 $(panel).prop("id" , aid?aid:panelIdPrefix+(id++));
518             }
519             $(panels[n]).omPanel('header').addClass( 'om-widget-header ');
