 15 (function( $, undefined ) {
 16      /**
 17       * @name omMessageTip
 18       * @author 周发志
 19       * @class
 20       * omMessageTip用于右下角弹出提示窗口（像QQ新闻一样）。<br/><br/>
 21       * <br/>
 22       * <h2>有以下特点：</h2><br/>
 23       * <ul>
 24       *     <li>不中断用户操作（非模态窗口提示）</li>
 25       *     <li>有较好的浏览器兼容性</li>
 26       *     <li>可以定义标题、内容，并且标题和内容可以使用html代码</li>
 27       *     <li>支持丰富的提示（图标不同）</li>
 28       *     <li>可以监听关闭事件</li>
 29       *     <li>弹出提示和关闭提示时有简单动画</li>
 30       *     <li>较轻量（仅简单的提示功能和定时消失功能，不可改变提示窗口大小，不可拖动提示窗口位置）</li>
 31       * </ul>
 32       * <br/>
 33       * 该组件非常轻量，功能也较少，如果需要中断用户操作，请使用omDialog或omMessageBox组件。内容区也仅能放html代码，如果有较复杂的内容请使用omDialog组件。
 34       * <br/>
 35       * <h2>提供了以下工具方法：</h2><br/>
 36       * <ul>
 37       *     <li>
 38       *         <b>$.omMessageTip.show(config)</b><br/>
 39       *         从当前页面右下角弹出一个非中断提示，弹出的提示可以关闭。其中config有以下配置项：<br/>
 40       *         <ul style="margin-left:40px">
 41       *             <li>type：提示的类型，类型不同时弹出窗口左边的图标会不同。String类型，可选的值有'alert'、'success'、'error'、'question'、'warning'、'waiting'。默认值为'alert'。</li>
 42       *             <li>title：弹出窗口的标题文字，String类型，可以使用普通字符串，也可以使用html代码。默认值为'提示'。</li>
 43       *             <li>content：弹出窗口的提示内容，String类型，可以使用普通字符串，也可以使用html代码。无默认值。</li>
 44       *             <li>onClose：弹出窗口关闭时的无参回调函数，Function类型。</li>
 45       *             <li>timeout：弹出窗口持续的时间，单位为毫秒，窗口弹出后经过这么长的时间后自动关闭（如果有onClose回调函数，会自动触发它），Int类型。默认值为无穷大（即不自动关闭）</li>
 46       *         </ul>
 47       *         <br/>使用方式如下：<br/>
 48       *         <pre>
 49       *             $.omMessageTip.show({
 50       *                 type:'warning',
 51       *                 title:'提醒',
 52       *                 content:'请选择你要删除的记录（可以选择一条或多条）！'
 53       *             });
 54       *             $.omMessageTip.show({
 55       *                 type:'error',
 56       *                 title:'数据非法',
 57       *                 content:'<font color="red">123456</font>不是有效的邮箱地址！',
 58       *                 onClose:function(){
 59       *                     $('#emial').focus();
 60       *                 }
 61       *             });
 62       *         </pre>
 63       *     </li>
 64       * </ul>
 65       */
 66     $.omMessageTip = {
 67         show: function(config){
 68             config = $.extend({
 69                 title : '提醒',
 70                 content : ' ',
 71                 type : 'alert'
 72             },config);
 73             var html = '<div class="om-messageTip om-widget om-corner-all" tabindex="-1">'+
 74                     '<div class="om-widget-header om-helper-clearfix">'+
 75                         '<span class="om-messageTip-title">'+config.title+'</span>'+
 76                         '<a href="#" class="om-messageTip-titlebar-close"><span class="om-icon om-icon-closethick"></span></a>' +
 77                     '</div>'+
 78                     '<div class="om-messageTip-content om-widget-content">'+
 79                         '<div class="om-messageTip-image om-messageTip-image-'+config.type+'"></div>' +
 80                         '<div class="om-messageTip-content-body">'+config.content+'</div>' +
 81                     '</div>'+
 82                 '</div>';
 83             var messageTip = $(html).appendTo(document.body).css('z-index', 10000).hide();
 84             var result = {d:messageTip,l:config.onClose};
 85             messageTip.find('a.om-messageTip-titlebar-close')
 86                 .bind('mouseenter mouseleave',function(){
 87                     $(this).toggleClass('om-state-hover');
 88                 })
 89                 .bind('focus blur',function(){
 90                     $(this).toggleClass('om-state-focus');
 91                 })
 92                 .bind('mousedown mouseup', function(){
 93                     $(this).toggleClass('om-state-mousedown');
 94                 })
 95                 .click(function(event){
 96                     $.omMessageTip._close(result);
 97                     return false;
 98                 });
 99             messageTip.slideDown('slow');
100             if(config.timeout){ //定时关闭
101                 setTimeout(function(){
102                     $.omMessageTip._close(result);
103                 },config.timeout);
104             }
105         },
106         _close : function(result){
107             result.d.slideUp('slow');
108             if(result.l){
109                 result.l(); //调用onClose回调函数
110             }
111             setTimeout(function(){
112                 result.d.remove();
113             },1000);
114         }
115     };
116 }(jQuery));
