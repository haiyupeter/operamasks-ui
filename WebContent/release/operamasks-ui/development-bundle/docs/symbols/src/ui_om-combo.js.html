658         _loadData:function(inputEl,records){
659             var combo = $.data(inputEl, 'combo');
660             var options = combo.options;
661             options.dataSource = records;
662             combo.dataHasLoaded = true;
663             //build all inputText
664             var inputField = options.inputField;
665             var allInputText = [];
666             if (typeof inputField === 'string') {
667                 $(records).each(function(){
668                     allInputText.push(this[inputField]);
669                 });
670             } else {
671                 $(records).each(function(index){
672                     allInputText.push(inputField(this, index));
673                 });
674             }
675             $.data(inputEl, 'allInputText', allInputText);
676             //build all value
677             var valueField = options.valueField;
678             var allValues = [];
679             if (typeof valueField === 'string') {
680                 $(records).each(function(){
681                     allValues.push(this[valueField]);
682                 });
683             } else {
684                 $(records).each(function(index){
685                     allValues.push(valueField(this, index));
686                 });
687             }
688             $.data(inputEl, 'allValues', allValues);
689             //build dropList
690             var dropList = $.data(inputEl, 'dropList').empty();
691             if (options.listProvider) {
692                 var selectableOptions = options.listProvider(dropList, records);
693                 if (selectableOptions) {
694                     selectableOptions.each(function(){
695                         $(this).addClass('om-combo-list-row');
696                     });
697                 }
698             } else {
699                 var optionField = options.optionField;
700                 var innerHtml = '';
701                 var self = this;
702                 if (typeof optionField === 'string') {
703                     $(records).each(function(index){
704                         innerHtml += self._wrapText(this[options.optionField]);
705                     });
706                 } else {
707                     $(records).each(function(index){
708                         innerHtml += self._wrapText(options.optionField(this, index));
709                     });
710                 }
711                 if (innerHtml) {
712                     $(innerHtml).appendTo(dropList);
713                     dropList.show().css('height','auto');
714                     if (options.listMaxHeight != 'auto' && dropList.height() > options.listMaxHeight) {
715                         dropList.height(options.listMaxHeight).css('overflow-y','auto');
716                     }
717                     dropList.hide();
718                     if(inputEl.parent().hasClass('om-state-hover')){
719                         self._showDropList(self.element);
720                     }
721                 }
722             }
723            
724             if (options.value) {
725                 this._setValue(inputEl, options.value);
726             }
727             this._bindEventsToList(inputEl);
728         },

831         _doFilter:function(){
832             var inputEl = this.element;
833             var options = $.data(inputEl, 'combo').options;
834             var records = options.dataSource;
835             var filterStrategy = options.filterStrategy;
836             var text = inputEl.val();
837             var items = this._getAllOptionsBeforeFiltered(inputEl);
838             var allInputText = $.data(inputEl, 'allInputText');
839             var self = this;
840             var needShow=false;
841             $(records).each(function(index){
842                 if (self._filetrPass(filterStrategy, text, records[index], allInputText[index])) {
843                     $(items.get(index)).removeClass('om-helper-hidden');
844                     needShow=true;
845                 } else {
846                     $(items.get(index)).addClass('om-helper-hidden');
847                 }
848             });
849             var dropList = $.data(inputEl, 'dropList').css('height','auto');
850             //过滤后重新计算下拉框的高度，看是否需要出现滚动条
851             if (options.listMaxHeight != 'auto' && dropList.height() > options.listMaxHeight) {
852                 dropList.height(options.listMaxHeight).css('overflow-y','auto');
853             }
854             if(!needShow){
855                 dropList.hide();
856             }
857         },
858         _filetrPass:function(filterStrategy,text,record,inputText){
859             if (text === '') {
860                 return true;
861             }
862             if (typeof filterStrategy === 'function') {
863                 return filterStrategy(text, record);
864             } else {
865                 if (filterStrategy === 'first') {
866                     return inputText.indexOf(text) === 0;
867                 } else if (filterStrategy === 'anywhere') {
868                     return inputText.indexOf(text) > -1;
869                 } else if (filterStrategy === 'last') {
870                     var i = inputText.lastIndexOf(text);
871                     return i > -1 && i + text.length == inputText.length;
872                 } else {
873                     return false;
874                 }
875             }
876         },

882             } else {
883                 if(inputEl.val() === emptyText){
884                     inputEl.val('');
885                 }
886                 inputEl.removeClass('om-empty-text');
887             }
