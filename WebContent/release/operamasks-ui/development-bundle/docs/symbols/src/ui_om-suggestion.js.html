 23      *     目前该组件主要用于从远程URL取得数据（如果是本地数据的话，可以使用omCombo组件，它也有边输入边过滤的功能）。一般用于从大量数据中进行查找的场合，如百度搜索、google搜索、taobao商品搜索、邮件系统快速输入收件人。<br/>
 24      *     该组件有客户端缓存的功能，如输入a开始Ajax查找；再输入b（输入框内容是ab）再次Ajax查找；再删除b（输入框内容是a）将不进行Ajax查找，因为缓存中已经有key=a的缓存内容，将直接根据缓存内容来重构可选列表而不发送Ajax请求从服务器取数。如果不需要缓存可以将cacheSize参数设成0。如果要清除缓存可以调用clearCache()方法。<br/>

481         _suggest:function(){
482             var inputEl = this.element;
483             var text = inputEl.val();
484             var last = $.data(inputEl, 'lastStr');
485             if (last && last === text) {
486                 return;
487             }
488             $.data(inputEl, 'lastStr', text);
489             var options = this.options;
490             var cache = $.data(inputEl, 'cache');
491             if (text.length > 0 && text.length >= options.minChars) {
492                 if (cache) {
493                     var data = cache[text];
494                     if (data) {//有缓存
495                         $.data(inputEl, 'records', data);
496                         this._buildDropList(data, text);
497                         return;
498                     }
499                 }
500                 //无缓存
501                 if (options.onBeforeSuggest) {
502                     if (options.onBeforeSuggest(text) === false) {
503                         $.data(inputEl, 'dropList').empty().hide();
504                         return;
505                     }
506                 }
507                 var self = this;
508                 var requestOption = {
509                     url: options.dataSource,
510                     type: options.method,
511                     dataType: options.crossDomain ? 'jsonp':'json',
512                     data: {},
513                     success: function(data, textStatus){
514                         var onSuccess = options.onSuccess;
515                         if (onSuccess && onSuccess(data, textStatus) === false) {
516                             return;
517                         }
518                         var preProcess = options.preProcess;
519                         if(preProcess){
520                             data = preProcess(text,data);
521                         }
522                         //如果有preProcess且没有返回值
523                         if(typeof data === 'undefined'){
524                             data=[];
525                         }
526                         //cache data
527                         if (options.cacheSize > 0) {
528                             var cache = $.data(inputEl, 'cache') ||
529                             {
530                                 ___keys: []
531                             };
532                             var keys = cache.___keys;
533                             if (keys.length == options.cacheSize) {
534                                 //cache满了先去掉一个
535                                 var k = keys[0];
536                                 cache.___keys = keys.slice(1);
537                                 cache[k] = undefined;
538                             }
539                             cache[text] = data;
540                             cache.___keys.push(text);
541                             $.data(inputEl, 'cache', cache);
542                         }
543                         $.data(inputEl, 'records', data);
544                         //buildDropList
545                         self._buildDropList(data, text);
546                     },
547                     error: function(XMLHttpRequest, textStatus, errorThrown){
548                         var onError = options.onError;
549                         if (onError) {
550                             onError(XMLHttpRequest, textStatus, errorThrown);
551                         }
552                     }
553                 };
554                 requestOption.data[options.queryName]=text;
555                 $.ajax(requestOption);
556                 var onSuggesting = options.onSuggesting;
557                 if (onSuggesting) {
558                     onSuggesting(text);
559                 }
560             } else {
561                 $.data(inputEl, 'dropList').empty().hide();
562             }
