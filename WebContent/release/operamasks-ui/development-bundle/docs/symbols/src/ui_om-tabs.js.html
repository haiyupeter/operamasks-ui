 95     function collectItems(self) {
 96         var options = $.data(self, 'omtabs').options;
 97         var items = [];
 98         $(self).find('>div.om-tabs-headers a').each(function(){
 99             var href  = this.getAttribute('href', 2);
100             var hrefBase = href.split( "#" )[ 0 ],
101                 baseEl;
102             if ( hrefBase && ( hrefBase === location.toString().split( "#" )[ 0 ] ||
103                     ( baseEl = $( "base" )[ 0 ]) && hrefBase === baseEl.href ) ) {
104                 href = this.hash;
105                 this.href = href;
106             }
107             var anchor = $(this);
108             var tabId = anchor.attr('tabId') || anchor.attr('id') || tabIdPrefix + id++ ;
109             anchor.attr('tabId', tabId);
110             var cfg = {
111                     tabId : tabId,
112                     title : anchor.text(),
113                     noHeader : true,
114                     onLoad : function() {
115                         options.onLoadComplete.call(self, tabId);
116                     },
117                     closed : true,//先全部隐藏.
118                     lazyLoad : true, //先全部不加载
119                     border : false
120             };
121             var target = $('>' + href, $(self))[0];
122             
123             // 考虑到tab DOM结构不完整的情况。
124             // 例如，当anchor的href='#tab-3'，而用户忘记在tabs里面写id=tab-3的DOM，此时不应该把#tab-3作为url进行load
125             // http://jira.apusic.net/browse/AOM-204
126             if (!target && href.indexOf('#') != 0) {
127                 //是url
128                 cfg.url = href;
129             }
130             var item = new OmPanel(target || $('<div></div>')[0], cfg);
131             items.push(item);
132         });
133         // items 是panel的集合.每一个item通过 $(item).omPanel('panel')之后能获取到对应的panel对象
134         return items;
135     }
136     function render(self) {
137         var data = $.data(self, 'omtabs'),
138             options = data.options,
139             omtabs = data.omtabs,
140             items = data.items;
141         // 对不合法的值处理
142         if(typeof options.active == 'number'){
143         	if (options.active < 0) {
144         		options.active = 0;
145         	}
146         	if (options.active > items.length - 1) {
147         		options.active = items.length - 1;
148         	}
149         }
150         if (options.width == 'fit') {
151             omtabs.css('width', '100%');
152         } else if (options.width != 'auto') {
153             omtabs.css('width', options.width);
154             // 解决IE7下，tabs在table>tr>td中ul把table的宽度撑宽的问题
155             omtabs.children(':first').css('width',options.width);
156         }
157         if (options.height == 'fit') {
158             omtabs.css('height', '100%');
159         } else if (options.height != 'auto') {
160             omtabs.css('height', options.height);
161         }
162         renderHeader(self);
163         renderBody(self);
164     }
165     function renderHeader(self) {
166         var data = $.data(self, 'omtabs'),
167             options = data.options,
168             omtabs = data.omtabs;
169         var headers = omtabs.find('>div.om-tabs-headers');
170         var lis = headers.find('ul li');
171         lis.addClass('om-state-default om-corner-top');
172         lis.each(function(n){
173             var innera = $(this).find('a:first');
174             if ($.browser.msie && parseInt($.browser.version) == 7 ) {
175                 innera.attr('hideFocus', 'true');
176             }
177             if (!innera.hasClass('om-tabs-inner')) {
178                 innera.addClass('om-tabs-inner');
179             }
180             if (n === options.active || options.active === innera.attr('tabId')) {
181                 $(this).addClass('om-tabs-activated om-state-active');
182                 options.activeTabId = innera.attr('tabId');
183                 options.active = n;
184             }
185             //tab width and height. by default, tabWidth=80 tabHeight=25, accept 'auto'
186             innera.css({
187                 'width' : options.tabWidth,
188                 'height' : options.tabHeight
189             });
190             if (options.closable===true || ($.isArray(options.closable) && -1 !== $.inArray(n,options.closable))) {
191                 $('<a class="om-icon om-icon-close"></a>').insertAfter(innera);
192             }
193         });
194         var aHeight = lis.find('a').height();
195         lis.parent().css({
196             'height' : ++ aHeight ,
197             'line-height' : aHeight + 'px'
198         });
199         _checkScroller(self) && _enableScroller(self);

201     function renderBody(self) {
202         var data = $.data(self, 'omtabs'),
203             options = data.options,
204             omtabs = data.omtabs,
205             items = data.items;
206         var panels = omtabs.find('>div.om-tabs-panels');
207         //detach all sub divs
208         panels.children().detach();
209         var styles = omtabs.attr('style'); 
210         if (styles && styles.indexOf('height') != -1 ) {
211             var omtabsHeight = omtabs.innerHeight(),
212                 headersHeight = omtabs.find('>div.om-tabs-headers').outerHeight();
213             panels.css('height', omtabsHeight - headersHeight);
214         }
215         if (!options.border) {
216             omtabs.css('border-width', '0');
217         }
218         var i = items.length;
219         while( i -- ) {
220             var panel = $(items[i]).omPanel('panel');
221             panel.prependTo(panels);
222         }
223     }

225         var data = $.data(self, 'omtabs'),
226             options = data.options,
227             items = data.items;
228         var i = items.length;
229         $(self).children().each(function(){
230             if (!$(this).hasClass('om-tabs-headers') &&

232                 $(this).remove();
233             }
234         });

240             if (i == options.active) {
241                 target.omPanel('open');
242             } else {
243                 target.omPanel('close');
244             }
245         }

247     function buildEvent(self) {
248         var data = $.data(self, 'omtabs'),
249             options = data.options,
250             omtabs = data.omtabs;
251         //close icon
252         omtabs.find('>div.om-tabs-headers a.om-icon-close').bind('click.omtabs', function(e){
253             var tabid = $(e.target).prev().attr('tabId');
254             _close(self, tabid);
255             return false;
256         });
257         // tab click
258         var tabInner = omtabs.find('>div.om-tabs-headers a.om-tabs-inner'); 
259         if (options.switchMode.indexOf('mouseover') != -1) {
260         	tabInner.bind('mouseover.omtabs', function() {
261                  var tabId = $(this).attr('tabId'), timer = $.data(self, 'activateTimer');
262                 (typeof timer !=='undefined') && clearTimeout(timer);
263                 timer = setTimeout(function(){
264                     _activate(self, tabId);
265                     return false;
266                 },500);
267                 $.data(self, 'activateTimer', timer);
268             });
269         } else if (options.switchMode.indexOf('click') != -1 ) {
270         	tabInner.bind('click.omtabs', function(){
271                 _activate(self, $(this).attr('tabId'));
272             });
273         }
274         tabInner.bind('click.omtabs',function(){
275         	return false;
276         });
277         if (options.autoPlay != false ) {
278             options.autoInterId = setInterval(function(){
279                 $(self).omTabs('activate', 'next');
280             }, options.interval);
281         }
282         //tab hover
283         if ( options.switchMode.indexOf("mouseover") == -1 ) {
284             var lis = omtabs.find('>div.om-tabs-headers li');
285             var addState = function( state, el ) {
286                 if ( el.is( ":not(.om-state-disabled)" ) ) {
287                     el.addClass( "om-state-" + state );
288                 }
289             };
290             var removeState = function( state, el ) {
291                 el.removeClass( "om-state-" + state );
292             };
293             lis.bind( "mouseover.omtabs" , function() {
294                 addState( "hover", $( this ) );
295             });
296             lis.bind( "mouseout.omtabs", function() {
297                 removeState( "hover", $( this ) );
298             });
299         }
300         //scroller click
301         omtabs.find('>div.om-tabs-headers >span').bind('click.omtabs', function(e) {
302             if ($(this).hasClass('om-tabs-scroll-disabled')) {
303                 return false;
304             }
305             var scrOffset =  $(this).offset();
306             var ul = $(this).parent().find('ul');
307             var li = ul.children(':last');
308             var dist = li.outerWidth(true);
309             var parent = $(this).parent();
310             if ($(this).hasClass('om-tabs-scroll-left')) {
311                 _scroll(self, dist, _scrollCbFn(self));
312             }
313             if ($(this).hasClass('om-tabs-scroll-right')) {
314                 _scroll(self, - dist, _scrollCbFn(self));
315             }
316             return false;
317         });
318     }

333     /**
334      * 选中特定的页签
335      * n 可为页签的索引（从0开始计数），或者页签的tabId TODO n 需要支持first  和 last 表示选中第一个和最后一个
336      */
337     function _activate(self, n) {
338         var data = $.data(self, 'omtabs'),
339             options = data.options,
340             omtabs = data.omtabs,
341             items = data.items;
342         var ul = omtabs.find('>div.om-tabs-headers ul');
343         if ( options.activeTabId == n || options.active == n ) {
344             return false;
345         }
346         n = n || 0;
347         var anchor , tid = n;
348         if ( n == 'next' ) {
349             n = (options.active + 1) % items.length ;
350         } else if ( n == 'prev' ) {
351             n = (options.active - 1) % items.length ;
352         } 
353         if (typeof n == 'number') {
354             tid = _getAlter(self, n);
355         } else if (typeof n == 'string') {
356             n = _getAlter(self, n);
357         }
358         if (options.onBeforeActivate.call(self, n) == false) {
359             return false;
360         }
361         anchor = ul.find('li a[tabId=' + tid + ']');
362         anchor.parent().siblings().removeClass('om-tabs-activated om-state-active');
363         anchor.parent().addClass('om-tabs-activated om-state-active');
364         options.activeTabId = tid;
365         options.active = n;
366         var i = items.length;
367         // 保证切换面板时先显示后隐藏，防止页面抖动的现象
368         for(i=items.length;i--;i>=0){
369         	var target = $(items[i]);
370         	if (target.omPanel('options').tabId == tid) {
371         		target.omPanel('open');
372         	}
373         }
374         for(i=items.length;i--;i>=0){
375         	var target = $(items[i]);
376         	if (target.omPanel('options').tabId != tid) {
377         		target.omPanel('close');
378         	}
379         }
380         //当选中了一个并未完全显示的页签,需要滚动让他完全显示出来
381         if (_checkScroller(self)) {
382             //stop every animation.
383             ul.stop(true, true);
384             $(self).clearQueue();
385             var lScroller = ul.prev();
386             var rScroller = ul.next();
387             var lBorder = anchor.parent().offset().left;
388             var rBorder = lBorder + anchor.parent().outerWidth(true);
389             var lDiff = lScroller.offset().left + lScroller.outerWidth(true) + 4 - lBorder ;
390             var rDiff = rScroller.offset().left - rBorder ;
391             if (lDiff >= 0) {
392                 _scroll(self, lDiff, _scrollCbFn(self));
393             } else if (rDiff <= 0) {
394                 _scroll(self, rDiff, _scrollCbFn(self));
395             } else {
396                 _scrollCbFn(self)();
397             }
398         }
399         options.onActivate.call(self, n);
400     }

402     /**
403      * 页签索引和tabId的转换器。
404      * 如果传入的id为数字，则表示页签的索引，函数返回页签的tabId；如果id为字符串，则表示该页签的tabId，函数返回页签的索引。
405      */
406     function _getAlter (self, id) {
407         var omtabs = $.data(self, 'omtabs').omtabs,
408             rt;
409         if (typeof id == 'number'){
410             rt = omtabs.find('>div.om-tabs-headers li:nth-child(' + ++id + ') a.om-tabs-inner').attr('tabId');
411         } else if (typeof id == 'string') {
412             omtabs.find('>div.om-tabs-headers li a.om-tabs-inner').each(function(i){
413                 if ($(this).attr('tabId') == id ) {
414                     rt = i;
415                     return false;
416                 }
417             });
418         }
419         return rt;
420     }

431     function _add (self, config/*title, content, url, closable , index,tabId*/) {
432         var data = $.data(self, 'omtabs'),
433             options = data.options,
434             omtabs = data.omtabs,
435             items = data.items;
436         var ul = omtabs.find('>div.om-tabs-headers ul');
437         var tabId = config.tabId?config.tabId:tabIdPrefix + id++;
438         //调整参数
439         config.index = config.index || 'last';
440         if (config.index == 'last' || config.index > items.length - 1) {
441             config.index = items.length;
442         }
443         config.title = config.title || 'New Title ' + tabId;
444         config.url = $.trim(config.url);
445         config.content = $.trim(config.content);
446         if (config.url) {
447             config.content = undefined;
448         } else {
449             config.url = undefined;
450             config.content = config.content || 'New Content ' + tabId;
451         }
452         if (options.onBeforeAdd.call(self, config/*title, content, url, closable , index*/) == false) {
453             return false;
454         }
455         var nHeader=$('<li class="om-state-default om-corner-top"> </li>');
456         var anchor = $('<a class="om-tabs-inner"></a>').html(config.title).attr({
457                 href : '#' + tabId,
458                 tabId : tabId
459             }).css({
460                 width : options.tabWidth,
461                 height : options.tabHeight
462             }).appendTo(nHeader);
463         if ($.browser.msie && parseInt($.browser.version) == 7) {
464             anchor.attr('hideFocus','true');
465         }
466         if ((config.closable === true) || 
467                 (config.closable == undefined && options.closable)) {
468             anchor.after('<a class="om-icon om-icon-close"></a>');
469         }
470         var cfg = {
471             tabId : tabId,
472             noHeader : true,
473             closed : true,
474             lazyLoad : options.lazyLoad,
475             onLoad : function() {
476                 options.onLoadComplete.call(self, tabId);
477             },
478             border : false
479         };
480         
481         $.extend(cfg, config);
482         var nPanel = new OmPanel($('<div></div>')[0],cfg);
483         if (config.index == items.length) {
484             items[config.index] = nPanel;
485             nHeader.appendTo(ul);
486         } else {
487             //insert at index
488             items.splice(config.index, 0, nPanel);
489             ul.children().eq(config.index).before(nHeader);
490         }
491         //every time we add or close an tab, check if scroller is needed.
492         _checkScroller(self) && _enableScroller(self);
493         renderBody(self);
494         purgeEvent(self);
495         buildEvent(self);
496         options.onAdd.call(self, cfg);
497         _activate(self, config.index);
498     }
499     /**
500      * 将index处的页签关闭，如果index指向当前页签，则激活下一页签；如果当前页签是最后一个页签，则激活第一个页签
501      * index :页签的位置，可为数字，tabId等。 TODO index将要支持prev，  next， first， last
502      */
503     function _close (self, index) {
504         var data = $.data(self, 'omtabs'),
505             options = data.options,
506             omtabs = data.omtabs,
507             items = data.items;
508         var headers = omtabs.find('>div.om-tabs-headers');
509         var panels = omtabs.find('>div.om-tabs-panels');
510         var omtabsHeight = omtabs.height();
511         index = index || options.active;
512         if (typeof index == 'string') {
513             //index is a tabid
514             index = _getAlter(self, index);
515         }
516         if (options.onBeforeClose.call(self, index) == false) {
517             return false;
518         }
519         headers.find('li').eq(index).remove();
520         panels.children().eq(index).remove();
521         items.splice(index, 1);
522         //in case of all tabs are closed, set body height
523         if (panels.children().length == 0) {
524             panels.css({height : omtabsHeight - headers.outerHeight()});
525         }
526         options.onClose.call(self, index);
527         if (items.length == 0) {
528             options.active = -1;
529             options.activeTabId = null;
530             return ;
531         } else if (index == options.active) {
532             options.active = -1;
533             !items[index] && (index = 0);
534             _activate(self, index);
535         } else {
536             index < options.active && options.active --;
537             _checkScroller(self) && _enableScroller(self);
538         }
539     }

543     function _closeAll(self) {

548         var headers = omtabs.find('>div.om-tabs-headers');

559         options.active = -1;
560         options.activeTabId = null;
561         options.onCloseAll.call(self);
562     }
563     /**
564      * 如果tab页签总宽度较大，则显示scroll并返回true；否则删除scroll并返回false。
565      */
566     function _checkScroller(self) {
567         var data = $.data(self, 'omtabs'),
568             options = data.options,
569             omtabs = data.omtabs;
570         if (!options.scrollable) {
571             return false;
572         }
573         var ul = omtabs.find('>div.om-tabs-headers ul');
574         var totalWidth = 0, flag = false;
575         if (ul.hasClass('om-tabs-scrollable')) {
576             //先假定没有左右滚动器 ,来计算宽度是否超过.
577             flag = true;
578             ul.removeClass('om-tabs-scrollable');
579         }
580         totalWidth += parseInt(ul.css('paddingLeft')) + parseInt(ul.css('paddingRight'));
581         if (flag == true) {
582             //重新加上滚动器 .
583             flag = false;
584             ul.addClass('om-tabs-scrollable');
585         }
586         ul.children().each(function() {
587             //计算一个li占用的总宽度
588             totalWidth += $(this).outerWidth(true);//sub element's width
589         });
590         if (totalWidth > ul.parent().innerWidth()) {
591             if (!ul.hasClass('om-tabs-scrollable')) {
592                 var leftScr = $('<span></span>').insertBefore(ul).addClass('om-tabs-scroll-left');
593                 var rightScr = $('<span></span>').insertAfter(ul).addClass('om-tabs-scroll-right');
594                 var mgn = (ul.height() - leftScr.height())/2;
595                 leftScr.add(rightScr).css({ // scroller in vertical center.
596                     'marginTop' : mgn,
597                     'marginBottom' : mgn
598                 });
599                 ul.addClass('om-tabs-scrollable');
600             }
601             return true;
602         } else {

616     /**
617      * 根据页签的位置，设置scroller的状态。
618      * 当最右边的页签顶住组件右边沿，则右边的scroller应该禁用，表示不能再往右滚动了。
619      * 当最左边的页签顶住组件左边沿，则左边的scroller应该禁用，表示不能再往左滚动了。
620      */
621     function _enableScroller (self) {
622         var omtabs = $.data(self, 'omtabs').omtabs;
623         var headers = omtabs.find('>div.om-tabs-headers');
624         var ul = headers.children('ul');
625         var lScroller = ul.prev();
626         var rScroller = ul.next();
627         var li = ul.children(':last');
628         var lBorder = headers.offset().left,
629             rBorder = rScroller.offset().left,
630             ulLeft = ul.offset().left,
631             ulRight = li.offset().left + li.outerWidth(true);
632         if (ulLeft < lBorder) {
633             lScroller.removeClass('om-tabs-scroll-disabled');
634         } else {
635             lScroller.addClass('om-tabs-scroll-disabled');
636             //_scroll(self, lBorder - ulLeft);
637         }
638         if (ulRight > rBorder) {
639             rScroller.removeClass('om-tabs-scroll-disabled');
640         } else {
641             rScroller.addClass('om-tabs-scroll-disabled');
642             //_scroll(self, rBorder - ulRight);
643         }
644     }
645     /**
646      * 将页签头部往右边滑动distance的距离。当distance为负数时，表示往左边滑动；fn为回调函数
647      */
648     function _scroll(self, distance, fn) {
649         var omtabs = $.data(self, 'omtabs').omtabs;
650         var ul = omtabs.find('>div.om-tabs-headers ul');
651         var li = ul.children(':last');
652         if (distance == 0) {
653             return;
654         }
655         var scrOffset = distance > 0 ? ul.prev().offset() : ul.next().offset();
656         var queuedFn = function(next) {
657             if (distance > 0 && ul.prev().hasClass('.om-tabs-scroll-disabled') ||
658                     distance < 0 && ul.next().hasClass('.om-tabs-scroll-disabled')){
659                 ul.stop(true, true);
660                 $(self).clearQueue();
661                 return;
662             }
663             var flag = false;
664             //fix distance.
665             distance = (distance > 0) ? '+=' + Math.min(scrOffset.left - ul.offset().left, distance) : 
666                 '-=' + Math.min(li.offset().left + li.outerWidth(true) - scrOffset.left, Math.abs(distance));
667             $.data(self, 'omtabs').isScrolling = true;
668             ul.animate({
669                 left : distance + 'px'
670             },'normal', 'swing', function() {
671                 !!fn && fn();
672                 $.data(self, 'omtabs').isScrolling = false;
673                 next();
674             });
675         };
676         $(self).queue(queuedFn);
677         if( $(self).queue().length == 1 && 
678                 !$.data(self, 'omtabs').isScrolling){
679             $(self).dequeue(); //start queue
680         }
681     }
