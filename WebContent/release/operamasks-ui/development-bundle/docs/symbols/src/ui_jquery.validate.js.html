 20  *  <pre>
 21  * 1、本校验框架提供了丰富的校验规则，满足常用的校验需求。
 22  *    内置的校验规则有required, min, max, minlength, maxlength, email,url,date,number, 
 23  *    creditcard,accept(验证可以输入的后缀),equalTo(和已有的属性做对比，对密码重新输入适用)

 26  *    对于空间比较吃紧的form表单，也可以采用鼠标移动到提示图标悬浮显示提示信息的方式。这些主要通过配置
 27  *    errorPlacement(定制错误信息显示)、showErrors(自定义消息处理，onblur/keyup触发，针对当前元素)
 28  *    只要你够强大，它就有多强大。
 29  *

 38  *         onkeyup : true,
 39  *         rules : {
 40  *           password : {
 41                         required : true,
 42                         minlength : 5
 43              }
 44  *         },
 45  *         messages : {
 46  *           password : {
 47                         required : '密码必须输入',
 48                         minlength : '密码不能小于5位'
 49              }
 50  *         }
 51  *      });
 52  *     
 53  *    上面就是基本的校验结构，reg是需要校验的form的id，password是需要校验的字段的name

 57  *      $.validator.addMethod("hasreg", function(value) {
 58  *              return value != 'admin';
 59  *       }, '此用户已经被注册');
 60  *       //模拟用户名是否已经被注册，真实的场景应该是使用ajax向后台发送请求，然后后台返回结果，

102 	validate: function( options ) {
103 
104 		// if nothing is selected, return nothing; can't chain anyway
105 		if (!this.length) {
106 			options && options.debug && window.console && console.warn( "nothing selected, can't validate, returning nothing" );
107 			return;
108 		}
109 
110 		// check if a validator for this form was already created
111 		var validator = $.data(this[0], 'validator');
112 		if ( validator ) {
113 			return validator;
114 		}
115 
116 		validator = new $.validator( options, this[0] );
117 		$.data(this[0], 'validator', validator);
118 
119 		if ( validator.settings.onsubmit ) {
120 
121 			// allow suppresing validation by adding a cancel class to the submit button
122 			this.find("input, button").filter(".cancel").click(function() {
123 				validator.cancelSubmit = true;
124 			});
125 
126 			// when a submitHandler is used, capture the submitting button
127 			if (validator.settings.submitHandler) {
128 				this.find("input, button").filter(":submit").click(function() {
129 					validator.submitButton = this;
130 				});
131 			}
132 
133 			// validate the form on submit
134 			this.submit( function( event ) {
135 				if ( validator.settings.debug )
136 					// prevent form submit to be able to see console output
137 					event.preventDefault();
138 
139 				function handle() {
140 					if ( validator.settings.submitHandler ) {
141 						if (validator.submitButton) {
142 							// insert a hidden input as a replacement for the missing submit button
143 							var hidden = $("<input type='hidden'/>").attr("name", validator.submitButton.name).val(validator.submitButton.value).appendTo(validator.currentForm);
144 						}
145 						validator.settings.submitHandler.call( validator, validator.currentForm );
146 						if (validator.submitButton) {
147 							// and clean up afterwards; thanks to no-block-scope, hidden can be referenced
148 							hidden.remove();
149 						}
150 						return false;
151 					}
152 					return true;
153 				}
154 
155 				// prevent submit for invalid forms or custom submit handlers
156 				if ( validator.cancelSubmit ) {
157 					validator.cancelSubmit = false;
158 					return handle();
159 				}
160 				if ( validator.form() ) {
161 					if ( validator.pendingRequest ) {
162 						validator.formSubmitted = true;
163 						return false;
164 					}
165 					return handle();
166 				} else {
167 					validator.focusInvalid();
168 					return false;
169 				}
170 			});
171 		}
172 
173 		return validator;
174 	},

177         if ( $(this[0]).is('form')) {
178             return this.validate().form();

184             });

186         }

196 		return result;

198 	// http://docs.jquery.com/Plugins/Validation/rules
199 	rules: function(command, argument) {
200 		var element = this[0];
201 
202 		if (command) {
203 			var settings = $.data(element.form, 'validator').settings;
204 			var staticRules = settings.rules;
205 			var existingRules = $.validator.staticRules(element);
206 			switch(command) {
207 			case "add":
208 				$.extend(existingRules, $.validator.normalizeRule(argument));
209 				staticRules[element.name] = existingRules;
210 				if (argument.messages)
211 					settings.messages[element.name] = $.extend( settings.messages[element.name], argument.messages );
212 				break;
213 			case "remove":
214 				if (!argument) {
215 					delete staticRules[element.name];
216 					return existingRules;
217 				}
218 				var filtered = {};
219 				$.each(argument.split(/\s/), function(index, method) {
220 					filtered[method] = existingRules[method];
221 					delete existingRules[method];
222 				});
223 				return filtered;
224 			}
225 		}
226 
227 		var data = $.validator.normalizeRules(
228 		$.extend(
229 			{},
230 			$.validator.metadataRules(element),
231 			$.validator.classRules(element),
232 			$.validator.attributeRules(element),
233 			$.validator.staticRules(element)
234 		), element);
235 
236 		// make sure required is at front
237 		if (data.required) {
238 			var param = data.required;
239 			delete data.required;
240 			data = $.extend({required: param}, data);
241 		}
242 
243 		return data;
244 	}

264 $.validator.format = function(source, params) {
265 	if ( arguments.length == 1 )
266 		return function() {
267 			var args = $.makeArray(arguments);
268 			args.unshift(source);
269 			return $.validator.format.apply( this, args );
270 		};
271 	if ( arguments.length > 2 && params.constructor != Array  ) {
272 		params = $.makeArray(arguments).slice(1);
273 	}
274 	if ( params.constructor != Array ) {
275 		params = [ params ];
276 	}
277 	$.each(params, function(i, n) {
278 		source = source.replace(new RegExp("\\{" + i + "\\}", "g"), n);
279 	});
280 	return source;
