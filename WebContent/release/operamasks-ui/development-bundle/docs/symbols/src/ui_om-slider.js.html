 12     /** 
 13      * @name omSlider
 14      * @author 李聪平
 15      * @class 用来展示页面中多个HTML元素的滑动器.<br/>
 16      * <b>特点：</b><br/>
 17      * <ol>
 18      * 		<li>以滑动器的方式展示页面中的多个元素，元素的HTML结构不限</li>
 19      * 		<li>内置控制导航条</li>
 20      * 		<li>内置多种切换的动画效果</li>
 21      * 		<li>可自定义导航条的内容和样式</li>
 22      * </ol>
 23      * <b>示例：</b><br/>
 24      * <pre>
 25      * <script type="text/javascript" >
 26      * $(document).ready(function() {
 27      *     $('#slider').omSlider({
 28      *         animSpeed : 100,
 29      *         effect : 'slide-v',
 30      *         onBeforeSlide : function(index){
 31      *             // do something
 32      *         }
 33      *     });
 34      * });
 35      * </script>
 36      * 
 37      * <div id="slider" class="slider-demo">
 38      *	<img src="images/turtle.jpg" />
 39      *	<a href="#"><img src="images/rabbit.jpg" /></a>
 40      *	<img src="images/penguin.jpg" />
 41      *	<img src="images/lizard.jpg" />
 42      *	<img src="images/crocodile.jpg" />
 43 	 * </div>
 44 	 * </pre>

 46      * @description 构造函数.

 50 	$.fn.omSlider = function(options) {
 51 		var methods = {
 52 	        /**
 53 	         * 切换到指定index的面板。
 54 	         * @name omSlider#slideTo
 55 	         * @function
 56 	         * @param index 面板的索引
 57 	         * @example
 58 	         * //切换到第三个面板
 59 	         * $('#slider').omSlider('slideTo', 2);
 60 	         */
 61 			slideTo : function(index) {
 62 				return this.each(function(){
 63 					opts = $(this).data('omSlider:opts');
 64 					_slideTo($(this),index);
 65 				});
 66 			},
 67 	        /**
 68 	         * 切换到下一个面板。
 69 	         * @name omSlider#next
 70 	         * @function
 71 	         * @example
 72 	         * $('#slider').omSlider('next');
 73 	         */			
 74 			next : function(){
 75 				return this.each(function(){
 76 					opts = $(this).data('omSlider:opts');
 77 					_next($(this));
 78 				});
 79 			},
 80 	        /**
 81 	         * 切换到上一个面板。
 82 	         * @name omSlider#prev
 83 	         * @function
 84 	         * @example
 85 	         * $('#slider').omSlider('prev');
 86 	         */			
 87 			prev : function(){
 88 				return this.each(function(){
 89 					opts = $(this).data('omSlider:opts');
 90 					_prev($(this));
 91 				});
 92 			}
 93 		};
 94 		if (methods[options]) {
 95 			return methods[options].apply(this, Array.prototype.slice.call(arguments, 1));
 96 		} 
 97 		var defaults = /** @lends omSlider#*/{
 98 	        /**
 99 	         * 设置面板是否自动切换。
100 	         * @default true
101 	         * @type Boolean
102 	         * @example
103 	         * $('#slider').omSlider({autoPlay : false});
104 	         */

108 	         * @default 5000
109 	         * @type Number
110 	         * @example
111 	         * $('#slider').omSlider({interval: 1000});//设置slider自动切换的间隔时间为1秒
112 	         */
113 			interval : 5000,
114 			/**
115 			 * 设置是否需要用来切换上一个或下一个面板的方向导航键，导航键在鼠标移动到slider上面的时候才出现。
116 			 * @default false
117 			 * @type Boolean
118 			 * @example
119 			 * $('#slider').omSlider({directionNav : true});
120 			 */
121 			directionNav: false,
122 			/**
123 			 * 设置当鼠标移动到slider上面的时候是否暂停自动切换。
124 			 * @default true
125 			 * @type Boolean
126 			 * @example
127 			 * $('#slider').omSlider({pauseOnHover : false});

129 			pauseOnHover: true,

131 			 * 设置是否需要导航条，当属性值为String的时候表示使用内置的导航条类型，
132 			 * 当属性值为Selector的时候表示使用自定义的导航条，当属性值设置为true的时候默认使用内置的"classical"导航条。
133 			 * 内置的导航条类型包括"classical"，"dot"。
134 			 * @default true
135 			 * @type Boolean,String,Selector
136 			 * @example
137 			 * $('#slider').omSlider({controlNav : false});//不使用导航条
138 			 * $('#slider').omSlider({controlNav : 'dot'});//使用内置的风格为'dot'的导航条
139 			 * $('#slider').omSlider({controlNav : 'div#my-nav'});//使用页面中id为'my-nav'的div作为导航条
140 			 */
141 			controlNav: true,
142 	        /**
143 	         * 设置导航条选中的时候设置的class样式，同时作用于内置导航条和自定义导航条。
144 	         * @default 'nav-selected'
145 	         * @type String
146 	         * @example
147 	         * $('#slider').omSlider({activeNavCls: 'my-nav-selected'});
148 	         */
149 			activeNavCls: 'nav-selected',

151 	        /**
152 	         * 设置面板切换的动画效果。
153 	         * 内置的动画效果包括'fade'(淡入淡出)、'slide-v'(垂直滑动)、'slide-h'(水平滑动)。
154 	         * 设置为true使用默认'fade'动画效果，设置为false不使用动画效果。
155 	         * @default 'fade'
156 	         * @type String,Boolean
157 	         * @example
158 	         * $('#slider').omSlider({effect : false});
159 	         * //使用垂直滑动的动画效果。
160 	         * $('#slider').omSlider({effect : 'slide-v'});
161 	         */
162 			effect : 'fade',
163 	        /**
164 	         * 动画执行的速度。单位毫秒，值越小动画执行的速度越快。
165 	         * @default 500
166 	         * @type Number
167 	         * @example
168 	         * $('#slider').omSlider({animSpeed : 100});
169 	         */
170 			animSpeed : 500,
171 			/**
172 			 * 组件初始化时默认激活的面板的index，index从0开始计算，0表示第一个面板。
173 			 * @default 0
174 			 * @type Number
175 	         * @example
176 	         * $('#slider').omSlider({startSlide : 2});
177 			 */
178 			startSlide: 0,
179 			/**
180 			 * 鼠标移动到导航条上面后触发切换动作的延迟时间。单位为毫秒。
181 			 * @default 200
182 			 * @type Number
183 	         * @example
184 	         * $('#slider').omSlider({delay : 100});
185 			 */
186 			delay: 200,
187             /**
188              * 面板切换前触发事件，事件的处理函数返回false则阻止切换动作。
189              * @event
190              * @type Function
191              * @default emptyFn
192              * @param index 面板的索引
193              * @name omSlider#onBeforeSlide
194 	         * @example
195 	         * $('#slider').omSlider({onBeforeSlide : function(index){if(index==2) return false;}});// 阻止slider切换到第三个面板
196              */
197 			onBeforeSlide:function(index){},
198 			/**
199 			 * 面板切换后触发事件。
200 			 * @event
201 			 * @type Function
202 			 * @default emptyFn
203 			 * @param index 面板的索引
204 			 * @name omSlider#onAfterSlide
205 			 * @example
206 			 * $('#slider').omSlider({onAfterSlide : function(index){alert(index + ' slide complete');});
207 			 */
208 			onAfterSlide:function(index){}

210 		var opts = $.extend({}, defaults, options);

212 		function _runSlideEffect(slider, index){
213 			var vars = slider.data('omSlider:vars'),
214 				$container = slider.find('ul.om-slider-content'),
215 				$item = $container.children(),
216 				top = 0,
217 				left = 0;
218 			if(opts.effect == 'slide-v'){
219 				// 垂直滑动效果
220 				$item.each(function(n){
221 					if(n == index) return false;
222 					top -= $(this).height();
223 				});
224 			} else if(opts.effect == 'slide-h'){
225 				// 水平滑动效果
226 				$item.each(function(n){
227 					if(n == index) return false;
228 					left -= $(this).width();
229 				});
230 			} else{
231 				return false;
232 			}
233 			vars.running = true;
234 			$container.stop().animate({top:top,left:left},opts.animSpeed,function(){
235 				vars.running = false;
236 				opts.onAfterSlide.call(self, index);
237 			});
238 		}
239 		
240 		function _runFadeEffect(slider,index){
241 			var vars = slider.data('omSlider:vars'),
242 				items = slider.find('ul.om-slider-content').children();
243 			items.each(function(n){
244 				var $child = $(this);
245 				if(n == index){
246 					vars.running = true;
247 					$child.fadeIn(opts.animSpeed,function(){
248 						vars.running = false;
249 						opts.onAfterSlide.call(self, index);
250 					});
251 				} else if(n == vars.currentSlide){
252 					$child.fadeOut(opts.animSpeed);
253 				}
254 			});
255 		}
256 		
257 		function _runNoEffect(slider,index){
258 			var vars = slider.data('omSlider:vars'),
259 				items = slider.find('ul.om-slider-content').children();
260 			items.each(function(n){
261 				var $child = $(this);
262 				if(n == index){
263 					$child.show();
264 					opts.onAfterSlide.call(self, index);
265 				} else if(n == vars.currentSlide){
266 					$child.hide();
267 				}
268 			});
269 		}
270 		
271 		/**
272 		 * 切换至指定面板，index从0开始
273 		 */
274 		function _slideTo(slider, index){
275 			var vars = slider.data('omSlider:vars');
276 			if(isNaN(index) || index < 0 || index >= vars.totalSlides){
277 				return;
278 			}
279 	        if (opts.onBeforeSlide.call(self, index) == false) {
280 	            return false;
281 	        }
282 			if(opts.effect == 'slide-h' || opts.effect == 'slide-v'){
283 				_runSlideEffect(slider, index);
284 			} else if(opts.effect == 'fade' || opts.effect === true){
285 				_runFadeEffect(slider, index);
286 			} else{
287 				_runNoEffect(slider, index);
288 			}
289 			
290 			if(vars.controlNav){
291 				var parent = slider;
292 				// 如果是自定义导航条，则controlNav的位置在slider外面，所以从body下面找controlNav
293 				if(vars.customNav){
294 					parent = $('body');
295 				}
296 				var navItems = parent.find(vars.controlNav).children();
297 				navItems.each(function(n){
298 					$(this).toggleClass(opts.activeNavCls,n==index);
299 				});
300 			}
301 			vars.currentSlide = index;
302 			return slider;
303 		}
304 		
305 		function _next(slider){
306 			var vars = slider.data('omSlider:vars'),
307 				next_index = 0;
308 			if(vars.currentSlide+2 <= vars.totalSlides){
309 				next_index = vars.currentSlide + 1;
310 			}
311 			return _slideTo(slider,next_index);
312 		}
313 		function _prev(slider){
314 			var vars = slider.data('omSlider:vars'),
315 				index = vars.totalSlides - 1;
316 			if(vars.currentSlide != 0){
317 				index = vars.currentSlide - 1;
318 			}
319 			return _slideTo(slider,index);
320 		}
321 		function _processDirectionNav(slider){
322 			var vars = slider.data('omSlider:vars'),
323 				directionNav = $('<div class="om-slider-directionNav">').appendTo(slider);
324 			$('<a class="om-slider-prevNav"></a>').appendTo(directionNav).click(function(){
325 				if(vars.running)return false;
326 				_prev(slider);
327 			});
328 			$('<a class="om-slider-nextNav"></a>').appendTo(directionNav).click(function(){
329 				if(vars.running)return false;
330 				_next(slider);
331 			});
332 			slider.hover(function(){
333 				directionNav.show();
334 			},function(){
335 				directionNav.hide();
336 			});
337 			
338 		} 
339 		function _processControlNav(slider){
340 			var vars = slider.data('omSlider:vars');
341 			if(opts.controlNav === true || opts.controlNav === 'classical'){
342 				var $nav = $('<ul class="om-slider-nav-classical"></ul>');
343 				vars.controlNav = '.om-slider-nav-classical';
344 				for(n=0;n<vars.totalSlides;n++){
345 					var $navItem = $('<li>'+(n+1)+'</li>');
346 					$navItem.data('sid',n);
347 					var hTimer = 0;
348 					$navItem.click(function(){
349 						//if(vars.running)return false;
350 						_slideTo(slider,$(this).data('sid'));
351 					});
352 					$navItem.hover(function(){
353 						if(vars.running)return false;
354 						var _self = $(this);
355 						if(_self.hasClass(opts.activeNavCls))return false;
356 						hTimer = setTimeout(function(){_slideTo(slider,_self.data('sid'));},opts.delay);
357 					},function(){
358 						clearTimeout(hTimer);
359 					});
360 					$nav.append($navItem);
361 				}
362 				slider.append($nav);
363 			} else if(opts.controlNav === 'dot'){
364 				var $nav = $('<div class="om-slider-nav-dot"></div>');
365 				vars.controlNav = '.om-slider-nav-dot';
366 				for(n=0;n<vars.totalSlides;n++){
367 					var $navItem = $('<a href="javascript:void(0)">'+(n+1)+'</a>');
368 					$navItem.data('sid',n);
369 					var hTimer = 0;
370 					$navItem.click(function(){
371 						//if(vars.running)return false;
372 						_slideTo(slider,$(this).data('sid'));
373 					});					
374 					$navItem.hover(function(){
375 						if(vars.running)return false;
376 						var _self = $(this);
377 						if(_self.hasClass(opts.activeNavCls))return false;
378 						hTimer = setTimeout(function(){_slideTo(slider,_self.data('sid'));},opts.delay);
379 					},function(){
380 						clearTimeout(hTimer);
381 					});
382 					$nav.append($navItem);
383 				}
384 				//$nav.insertAfter(slider);
385 				$nav.appendTo(slider).css({marginLeft:-1*$nav.width()/2});
386 			} else{
387 				if($(opts.controlNav).length > 0){
388 					vars.controlNav = opts.controlNav;
389 					vars.customNav = true;
390 					var $nav = $(opts.controlNav);
391 					$nav.children().each(function(n){
392 						var $navItem = $(this);
393 						$navItem.data('sid',n);
394 						var hTimer = 0;
395 						$navItem.click(function(){
396 							//if(vars.running)return false;
397 							_slideTo(slider,$(this).data('sid'));
398 						});
399 						$navItem.hover(function(){
400 							if(vars.running)return false;
401 							var _self = $(this);
402 							if(_self.hasClass(opts.activeNavCls))return false;
403 							hTimer = setTimeout(function(){_slideTo(slider,_self.data('sid'));},opts.delay);
404 						},function(){
405 							clearTimeout(hTimer);
406 						});
407 					});
408 				}
409 			}
410 		}
411 		return this.each(function() {
412 			var timer = 0;
413 			var $this = $(this);
414 	        var vars = {
415                 currentSlide: 0,
416                 totalSlides: 0,
417                 running: false,
418                 paused: false,
419                 stop: false,
420                 controlNav: '.om-slider-nav-classical'
421             };
422 			var data = $this.data('omSlider');
423 			if(data){return data;}
424 			$this.data('omSlider',$this);
425 			$this.data('omSlider:vars',vars);
426 			$this.data('omSlider:opts',opts);
427 			$this.addClass('om-slider');
428 			if(opts.startSlide > 0){
429 				vars.currentSlide = opts.startSlide; 
430 			}
431 			var kids = $this.children();
432 			kids.wrapAll('<ul class="om-slider-content"></ul>').wrap('<li class="om-slider-item"></li>');
433 			if(opts.effect == 'slide-v' || opts.effect == 'slide-h'){
434 				$this.find('.om-slider-content').addClass('om-slider-effect-'+opts.effect);
435 			}
436 			vars.totalSlides = kids.length;
437 			
438 			_processControlNav($this);
439 			_slideTo($this,vars.currentSlide);
440 			if(opts.autoPlay){
441 				timer = setInterval(function(){_next($this);},opts.interval);
442 			}
443 			if(opts.pauseOnHover){
444 				$this.hover(function(){
445 					vars.paused = true;
446 					clearInterval(timer);
447 				},function(){
448 					vars.paused = false;
449 					if(opts.autoPlay){
450 						timer = setInterval(function(){_next($this);},opts.interval);
451 					}
452 				});
453 			}
454 			if(opts.directionNav){
455 				_processDirectionNav($this);
456 			}
457 		});
458 	};
