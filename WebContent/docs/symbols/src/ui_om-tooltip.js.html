 14 ;(function($) {
 15     /**
 16      * @name omTooltip
 17      * @class 提示组件，当某个链接、表单、输入框等需要做功能向导提示的时候可以使用本组件。
 18      * <b>特点：</b><br/>
 19      * <ol>
 20      *      <li>提示内容多样化，可以是文字，html，页面dom节点还可以是异步加载的内容</li>
 21      *      <li>异步加载内容的时候支持懒加载</li>
 22      *      <li>最大限度节约资源，只有需要使用的时候才在页面加入dom元素</li>
 23      *      <li>支持所有标准html元素</li>
 24      * </ol>
 25      * 
 26      * <b>示例：</b><br/>
 27      * <pre>
 28      * <script type="text/javascript" >
 29      * $(document).ready(function() {
 30      *     $('#tip').omTooltip({
 31      *         trackMouse : true,
 32      *         html : '<div style="color:red;">欢迎使用omTooltip组件</div>'
 33      *     });
 34      * });
 35      * </script>
 36      * 
 37      * <input id="tip" type="submit" />
 38      * </pre>
 39      * 
 40      * @constructor
 41      * @description 构造函数. 
 42      * @param p 标准config对象：{}
 43      */
 44     $.omWidget('om.omTooltip', {
 45         options: /**@lends omTooltip# */{
 46             /**
 47              * 提示框的宽度。
 48              * @type Number 
 49              * @default 'auto' 
 50              * @example
 51              * $("#tip").omTooltip({width: 100}); 
 52              */
 53             width : 'auto',
 54             /**
 55              * 最小宽度，当提示内容很少的时候为了样式美观显示指定的宽度。
 56              * @type Number 
 57              * @default null 
 58              * @example
 59              * $("#tip").omTooltip({minWidth: 100}); 
 60              */
 61             minWidth : null,
 62             /**
 63              * 最大宽度，当需要根据内容自动适应宽度而又不想因为内容太多而导致太宽的情况，可以使用maxWidht设置一个最大宽度。
 64              * @type Number 
 65              * @default null 
 66              * @example
 67              * $("#tip").omTooltip({maxWidth: 200}); 
 68              */
 69             maxWidth : null,
 70             /**
 71              * 提示框的高度。
 72              * @type Number 
 73              * @default 'auto' 
 74              * @example
 75              * $("#tip").omTooltip({height: 100}); 
 76              */
 77             height : 'auto',
 78             /**
 79              * 最大高度，高度默认自适应，当达到最大高度之后会隐藏内容 。
 80              * @type Number 
 81              * @default null 
 82              * @example
 83              * $("#tip").omTooltip({maxHeight: 100}); 
 84              */
 85             maxHeight : null,
 86             /**
 87              * 最小高度。
 88              * @type Number 
 89              * @default null 
 90              * @example
 91              * $("#tip").omTooltip({minHeight: 100}); 
 92              */
 93             minHeight : null,
 94             /**
 95              * 鼠标移入目标区域之后弹出提示框的延迟时间(ms) ，同时也是隐藏延迟的事件。
 96              * @type Number 
 97              * @default 300 
 98              * @example
 99              * $("#tip").omTooltip({delay: 300}); 
100              */
101             delay : 300,
102             /**
103              * 是否显示anchor，默认不显示，如果trackMouse为true，则显示在左上角，如果不跟随，
104              * 则根据region属性确定，箭头的方向永远指向被提示的目标。
105              * @type Boolean 
106              * @default false 
107              * @example
108              * $("#tip").omTooltip({anchor: false}); 
109              */
110             anchor : false,
111             /**
112              * 提示框是否跟随鼠标移动,默认不跟随鼠标移动，设置为true则跟随鼠标移动，
113              * 比region优先级低。
114              * @type Boolean 
115              * @default false 
116              * @example
117              * $("#tip").omTooltip({trackMouse: false}); 
118              */
119             trackMouse : false,
120             /**
121              * 显示提示框的触发方式,可选值有mouseover、click 
122              * @type String 
123              * @default 'mouseover' 
124              * @example
125              * $("#tip").omTooltip({showOn: click}); 
126              */
127             showOn : "mouseover",
128             /**
129              * ajax方式显示内容的时候是否延迟加载页面，必须配置了url才有效 
130              * @type Boolean 
131              * @default false 
132              * @example
133              * $("#tip").omTooltip({url:'a.html',lazyLoad: true}); 
134              */
135             lazyLoad : false,
136             /**
137              * ajax加载内容的url地址,返回的内容格式dataType为html
138              * @type String 
139              * @default null 
140              * @example
141              * $("#tip").omTooltip({url:'a.html'}); 
142              */
143             url : null,
144             /**
145              * 提示框显示在指定位置的偏离度,提示框有两种基准位置，一是鼠标，二是目标区域,
146              *  第一个参数为top偏差值，第二个为left偏差值。
147              * @type Object 
148              * @default [5,5] 
149              * @example
150              * $("#tip").omTooltip({offset:[10,15]}); 
151              */
152             offset : [5,5],
153             /**
154              * anchor偏移度(px),
155              * 当设置anchor为top、button的时候，偏移度指相对于最左边的距离；
156              * 为left、right的时候偏移度指相对于最上的距离
157              * @type Number
158              * @default 0
159              * @example
160              * $("#tip").omTooltip({anchorOffset:10}); 
161              */
162             anchorOffset : 0,
163             /**
164              * 提示框显示的区域,设置提示框显示的区域，如果不设置，
165              * 则以鼠标的当前位置为基准点显示并跟随鼠标移动，
166              * 如果设置了则以目标为基准点显示,可选值有left、right、top、bottom，
167              * 比如选择了left，则提示框永远出现在目标的左边，不会进入目标。
168              * @type String
169              * @default null
170              * @example
171              * $("#tip").omTooltip({region:'bottom'}); 
172              */
173             region : null,
174             /**
175              * 提示组件显示的内容，可以是html和普通字符,优先级比contentEL属性高。
176              * @type String
177              * @default null
178              * @example
179              * $("#tip").omTooltip({html:'<span style="color:red;">我是omTooltip</span>'}); 
180              */
181             html : null,
182             /**
183              * jquery选择器，可以获取页面的dom元素作为提示内容，格式为#ID、.className、tagName等，优先级比html属性低。
184              * @type selector
185              * @default null
186              * @example
187              * $("#tip").omTooltip({contentEL:'#aa'}); 
188              */
189             contentEL : null
190             
191         },
192         _create:function(){
193             this.tipContent = $('<div class="tip-body"></div>');
194             this.tipAnchor = $('<div class="tip-anchor"></div>');
195             this.regionMap = {'right':'left','top':'bottom','left':'right','bottom':'top'};
196             this.tip = $('<div class="tip"></div>').append(this.tipContent);
197         },
198         /**
199          * 初始化方法，主要设置高宽，加载内容
200          */
201         _init:function(){
202             var self = this , options = self.options;
203             
204             this.tipContent.empty();
205             this.tip.find('div.tip-anchor').remove();
206             
207             if(options.anchor){
208                 self.tip.append(self.tipAnchor);
209             }
210             self.adjustWidth = 6 ; //内外边框
211             self.adjustHeight =  6 ;
212             if(options.width != 'auto'){
213                 this.tip.css('width',options.width-self.adjustWidth);
214             }else{
215                 this.tip.css('width','auto');
216             }
217             if(options.height != 'auto'){
218                 this.tip.css('height',options.height - self.adjustHeight);
219             }else{
220                 this.tip.css('height','auto');
221             }
222             if(options.minWidth) {
223                 this.tip.css('minWidth',options.minWidth - self.adjustWidth);
224             }
225             if(options.maxWidth){
226                 this.tip.css('maxWidth',options.maxWidth - self.adjustWidth);
227             }
228             if(options.maxHeight){
229                 this.tip.css('maxHeight',options.maxHeight - self.adjustHeight);
230             }
231             if(options.minHeight){
232                 this.tip.css('minHeight',options.minHeight - self.adjustHeight);
233             }
234             /**
235              * 内容加载的优先级为url、html、contentEl，前者会覆盖后者
236              */
237             if(options.url && !options.lazyLoad){ //如果是ajax方式并且不是懒加载模式下面，初始化组件的时候就请求数据
238                 self._ajaxLoad(options.url);
239             }else if(options.html){
240                 this.tip.find('.tip-body').append(options.html);
241             }else{
242                 this.tip.find('.tip-body').append($(options.contentEL).html());
243             }
244             this._bindEvent();
245         },
246         /**
247          * ajax方法异步加载数据，返回的数据格式为html。
248          * @param url
249          */
250         _ajaxLoad:function(url){
251             var self = this;
252             $.ajax({
253                 url: url,
254                 method: 'POST',
255                 dataType: 'html',
256                 success: function(data, textStatus){
257                     self.tip.find('.tip-body').append(data);
258                 },
259                 error: function(XMLHttpRequest, textStatus, errorThrown){
260                     self.tip.find('.tip-body').append($.om.lang.omTooltip.FailedToLoad);
261                 }
262             });
263         },
264         /**
265          * 给提示框绑定事件，同时根据配置的delay参数做延迟显示，延迟隐藏处理
266          * 而且鼠标离开了目标区域，但是还停留在提示框内的时候不能隐藏
267          */
268         _bindEvent : function(){
269             var self = this , options = self.options , element = self.element;
270             if(options.showOn == 'mouseover'){ //mouseover的时候显示
271                 element.bind('mouseover',function(e){
272                     if(self.showTime)clearTimeout(self.showTime); 
273                     self.showTime = setTimeout(function(){
274                         self.show(e);
275                     },options.delay);
276                 });
277             }else if(options.showOn == 'click'){ //click的时候显示
278                 self.showTime = element.bind('click',function(e){
279                     setTimeout(function(){
280                         self.show(e);
281                     },options.delay);
282                 });
283             }
284             if(options.trackMouse){
285                 element.bind('mousemove',function(e){
286                     self._adjustPosition(e);
287                 });
288             }
289             element.bind('mouseleave',function(){
290                 if(self.showTime)clearTimeout(self.showTime); 
291                 self.hideTime = setTimeout(function(){ self.hide(); }, options.delay);
292             });
293             self.tip.bind('mouseover',function(){
294                 if(self.hideTime){clearTimeout(self.hideTime);}
295             }).bind('mouseleave',function(){
296                 setTimeout(function(){ self.hide(); }, options.delay);
297             });
298         },
299         /**
300          * 显示组件，组件只有在显示的时候才添加到dom树，
301          * 第二次显示已经存在dom树，只需要改变显示状态。
302          * @name omTooltip#show
303          * @function
304          * @example
305          * $('#mytip').omTooltip('show');
306          */
307         show : function(e){
308             var self = this ,options = self.options;
309             if($(document.body).find(self.tip).length <= 0){ //组件没有写入dom树的情况下
310                 if(options.url && options.lazyLoad){
311                     self._ajaxLoad(options.url);
312                 }
313                 self.tip.appendTo(document.body).fadeIn();
314                 self._adjustPosition(e);
315             }else{
316                 self._adjustPosition(e);
317                 self.tip.fadeIn();
318             }
319         },
320         /**
321          * 调整提示框的位置，如果传入了event且trackMouse为true，则会根据event计算出鼠标的位置
322          * 作为提示框的基础位置，再根据设置的offset等计算最终的提示框位置
323          * 如果没有传入event或者trackMouse为false，则提示框将以目标位置为基础位置，再根据offset
324          * region等参数配置情况计算最终位置。
325          * 这里会根据浏览器的宽度对显示做调整，如果region配置了right，二浏览器右边已经没有足够的空间
326          * 容纳提示框，则会将提示框显示到左边，region配置了bottom的时候同理。
327          * @param event
328          */
329         _adjustPosition : function(event){
330             var self = this , options = self.options , 
331             element = self.element , top ,left,
332             offSet = $(element).offset();
333            if(event && !options.region){
334                 top = event.pageY + options.offset[0]; 
335                 left = event.pageX + options.offset[1]; 
336                 //anchor距离最顶部的距离，使用用户设置的偏移减去tip的高度和调节高度
337                 self.tip.find('.tip-anchor').css({'top':options.anchorOffset+3,'left':'-9px'}).addClass('tip-anchor-left');
338            }else{
339                var bottomWidth = parseInt($(element).css('borderBottomWidth').replace('px',''));
340                top = offSet.top +  $(element).height() + (isNaN(bottomWidth)?0:bottomWidth) +options.offset[0]; //1px作为调节距离
341                left = offSet.left +options.offset[1];
342                //-----------根据region配置显示固定位置--------------
343                if(options.region == 'right'){ //offset向右下偏 right作为默认配置
344                    left = offSet.left + options.offset[1] + $(element).width() + self.adjustWidth; //元素的offset向右移动$(element).widht()的距离
345                    top = top - self.element.height() + self.adjustHeight + options.offset[0]; 
346                    self.tip.find('.tip-anchor').css({'top':options.anchorOffset+3,'left':'-9px'}); //要减去自身的高度,去掉3px的radius
347                }else if(options.region == 'left'){ //offset向左上偏
348                    left = offSet.left - self.tip.width() - self.adjustWidth - options.offset[1] - 2; //靠左的话就要减去tip自身的宽度
349                    top = top - self.element.height() - self.adjustHeight - options.offset[0];
350                    self.tip.find('.tip-anchor').css({'top':options.anchorOffset+3,'left': self.tip.outerWidth()-2}); //将anchor向右移动tip的宽度
351                }else if(options.region == 'top'){ //offset向右上偏
352                    top = top - (self.element.height() + self.tip.height() + self.adjustHeight + 2) - options.offset[0];
353                    left = left + options.offset[1];
354                    self.tip.find('.tip-anchor').css({'top':self.tip.outerHeight()-3, 'left':options.anchorOffset+3}); //25+4
355                }else if(options.region == 'bottom'){ //offset向右下偏
356                    top = top + options.offset[0];
357                    left = left + options.offset[1];
358                    self.tip.find('.tip-anchor').css({'top': -9, 'left':options.anchorOffset + 3});
359                }
360                self.tip.find('.tip-anchor').addClass('tip-anchor-'+self.regionMap[options.region]);
361            }
362            if((left + self.tip.width()) > document.documentElement.clientWidth){ //当右边距离过短的时候会将提示框调整到左边
363                left = left - self.tip.width() - 20;
364            }
365            if((top + self.tip.height()) > document.documentElement.clientHeight){ //当下边距离过短的时候会将提示框调整到上边
366                top = top - (self.element.height() + self.tip.height()) - 20;
367            }
368            self.tip.css({'top':top - $(document).scrollTop(),'left':left - $(document).scrollLeft()});
369         },
370         /**
371          * 隐藏组件。
372          * @name omTooltip#hide
373          * @function
374          * @example
375          * $('#mytip').omTooltip('hide');
376          */
377         hide : function(){
378             this.tip.hide();
379         },
380         destroy : function(){
381         	clearTimeout(this.showTime);
382         	clearTimeout(this.hideTime);
383         	this.tip.remove();
384         }
385     });
386     $.om.lang.omTooltip = {
387             FailedToLoad : '加载提示信息出错！'
388     };
389 })(jQuery);
