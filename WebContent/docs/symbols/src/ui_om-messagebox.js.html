 19 	 var tmpl = '<div class="om-messageBox om-widget om-widget-content om-corner-all" tabindex="-1">'+
 20 	                '<div class="om-messageBox-titlebar om-widget-header om-helper-clearfix">'+
 21 	                    '<span class="om-messageBox-title"></span>'+

 23 	                '</div>'+
 24 	                '<div class="om-messageBox-content om-widget-content">'+
 25 	                    '<table><tr vailgn="top">' +
 26 	                        '<td class="om-messageBox-imageTd"><div class="om-messageBox-image"/> </td>' +
 27 	                        '<td class="om-message-content-html"></td>' +
 28 	                    '</tr></table>'+
 29 	                '</div>'+
 30 	                '<div class="om-messageBox-buttonpane om-widget-content om-helper-clearfix">'+
 31 	                    '<div class="om-messageBox-buttonset"></div>'+
 32 	                '</div>'+
 33 	            '</div>';
 34 	var _height = function(){
 35         // handle IE 6
 36         if ($.browser.msie && $.browser.version < 7) {
 37             var scrollHeight = Math.max(document.documentElement.scrollHeight, document.body.scrollHeight),
 38                 offsetHeight = Math.max(document.documentElement.offsetHeight, document.body.offsetHeight);
 39             return (scrollHeight < offsetHeight) ?  $(window).height() : scrollHeight;
 40         // handle "good" browsers
 41         } else {
 42             return $(document).height();
 43         }
 44 	};
 45 	var _width = function() {
 46         // handle IE
 47         if ( $.browser.msie ) {
 48             var scrollWidth = Math.max(document.documentElement.scrollWidth, document.body.scrollWidth),
 49                 offsetWidth = Math.max(document.documentElement.offsetWidth, document.body.offsetWidth);
 50             return (scrollWidth < offsetWidth) ? $(window).width() : scrollWidth;
 51         // handle "good" browsers
 52         } else {
 53             return $(document).width();
 54         }
 55     };
 56 	var close = function(messageBox, mask, handler, value){
 57 	    if (messageBox.hasClass('om-messageBox-waiting')) {
 58 	        return;
 59 	    }
 60 	    handler ? handler(value) : jQuery.noop();
 61 	    messageBox.remove();
 62 	    mask.remove();
 63 	};
 64     var _show = function(config){
 65         var onClose = config.onClose;
 66         var messageBox = $(tmpl).appendTo(document.body).css('z-index', 1500).position({
 67             of:window,
 68             collision: 'fit'
 69         }).omDraggable({
 70             containment: 'document',
 71             cursor:'move',
 72             handle: '.om-messageBox-titlebar'
 73         }).hide().keydown(function(event){
 74             if (event.keyCode && event.keyCode === $.om.keyCode.ESCAPE) {
 75                 close(messageBox, mask, null, false);
 76                 event.preventDefault();
 77             }
 78         });
 79         var mask = $('<div class="om-widget-overlay"/>').appendTo(document.body).show().css({height:_height(),width:_width()});
 80         var closeBut = messageBox.find('span.om-messageBox-title').html(config.title).next().hover(function(){
 81             $(this).addClass('om-state-hover');
 82         }, function(){
 83             $(this).removeClass('om-state-hover');
 84         }).focus(function(){
 85             $(this).addClass('om-state-focus');
 86         }).blur(function(){
 87             $(this).removeClass('om-state-focus');
 88         }).click(function(event){
 89             close(messageBox, mask, null, false);
 90             return false;
 91         }).bind('mousedown mouseup', function(){
 92             $(this).toggleClass('om-state-mousedown');
 93         });
 94         messageBox.find('div.om-messageBox-image').addClass('om-messageBox-image-' + config.type);
 95         var content = config.content;
 96         if (config.type == 'prompt') {
 97             content = content || '';
 98             content += '<br/><input id="om-messageBox-prompt-input" type="text"/>';
 99         }
100         messageBox.find('td.om-message-content-html').html(content);
101         var buttonSet = messageBox.find('div.om-messageBox-buttonset');
102         switch (config.type) {
103             case 'confirm':
104                 buttonSet.html('<button>确定</button><button>取消</button>').children().first().click(function(){
105                     close(messageBox, mask, onClose, true);
106                 }).next().click(function(){
107                     close(messageBox, mask, onClose, false);
108                 });
109                 break;
110             case 'prompt':
111                 buttonSet.html('<button>确定</button><button>取消</button>').children().first().click(function(){
112                     var returnValue = onClose ? onClose($('#om-messageBox-prompt-input').val()) : jQuery.noop();
113                     if (returnValue !== false) {
114                         messageBox.remove();
115                         mask.remove();
116                     }
117                 }).next().click(function(){
118                     close(messageBox, mask, onClose, false);
119                 });
120                 break;
121             case 'waiting':
122                 messageBox.addClass('om-messageBox-waiting');
123                 mask.addClass('om-messageBox-waiting');
124                 closeBut.hide(); //不显示关闭按钮
125                 buttonSet.parent().hide(); //不显示下面的按钮面板
126                 break;
127             default:
128                 buttonSet.html('<button>确定</button>').children().first().click(function(){
129                     close(messageBox, mask, onClose, true);
130                 });
131         }
132         var buts = $('button',buttonSet);
133         if($.fn.omButton){
134             buts.omButton();
135         }
136         messageBox.show();
137         var okBut = buts.first()[0];
138         okBut ? okBut.focus() : messageBox.focus();
139     };
140      /**
141       * @name omMessageBox
142       * @class
143       * omMessageBox用于提供提示信息的弹出窗口，类似于JavaScript中使用alert()、confirm()、prompt()函数时出现的那种提示信息的弹出窗口。<br/><br/>
144       * <br/>
145       * <h2>有以下特点：</h2><br/>
146       * <ul>
147       *     <li>有较好的浏览器兼容性</li>
148       *     <li>可以定义标题、内容，并且标题和内容可以使用html代码</li>
149       *     <li>标题栏有关闭按钮，也可以按Esc键关闭</li>
150       *     <li>支持丰富的提示（图标不同）</li>
151       *     <li>可以监听关闭事件</li>
152       * </ul>
153       * <br/>
154       * <h2>提供了以下工具方法：</h2><br/>
155       * <ul>
156       *     <li>
157       *         <b>$.omMessageBox.alert(config)</b><br/>
158       *         弹出一个Alert提示，仅有一个“确定”按钮。其中config有以下配置项：<br/>
159       *         <ul style="margin-left:40px">
160       *             <li>type：alert提示的类型，类型不同时弹出窗口左边的图标会不同。String类型，可选的值有'alert'、'success'、'error'、'question'、'warning'。默认值为'alert'。</li>
161       *             <li>title：弹出窗口的标题文字，String类型，可以使用普通字符串，也可以使用html代码。默认值为'提示'。</li>
162       *             <li>content：弹出窗口的提示内容，String类型，可以使用普通字符串，也可以使用html代码。无默认值。</li>
163       *             <li>onClose：弹出窗口关闭时的回调函数，Function类型，点击"确定"按钮来关闭弹出窗口时，Function的参数value值为true，按ESC键关闭弹出窗口时，Function的参数value值为false。无默认值。</li>
164       *         </ul>
165       *         <br/>使用方式如下：<br/>
166       *         <pre>
167       *             $.omMessageBox.alert({
168       *                 type:'error',
169       *                 title:'失败',
170       *                 content:'不能删除<font color="red">admin</font>用户',
171       *                 onClose:function(value){
172       *                     alert('do something');
173       *                 }
174       *             });
175       *         </pre>
176       *     </li>
177       *     <li>
178       *         <b>$.omMessageBox.confirm(config)</b><br/>
179       *         弹出一个Confirm提示，有“确定”和“取消”按钮。其中config有以下配置项：<br/>
180       *         <ul style="margin-left:40px">
181       *             <li>title：弹出窗口的标题文字，String类型，可以使用普通字符串，也可以使用html代码。默认值为'确认'。</li>
182       *             <li>content：弹出窗口的提示内容，String类型，可以使用普通字符串，也可以使用html代码。无默认值。</li>
183       *             <li>onClose：弹出窗口关闭时的回调函数，Function类型，点击"确定"按钮来关闭弹出窗口时，Function的参数value值为true，点击“取消”按钮或按ESC键关闭弹出窗口时，Function的参数value值为false。无默认值。</li>
184       *         </ul>
185       *         <br/>使用方式如下：<br/>
186       *         <pre>
187       *             $.omMessageBox.confirm({
188       *                 title:'确认删除',
189       *                 content:'删除用户后，它所有的发帖和回帖将同时删除（不可恢复），你确定要这样做吗？',
190       *                 onClose:function(value){
191       *                     alert(value?'开始删除操作':'不删除了');
192       *                 }
193       *             });
194       *         </pre>
195       *     </li>
196       *     <li>
197       *         <b>$.omMessageBox.prompt(config)</b><br/>
198       *         弹出一个Prompt提示，有一个输入框和“确定”和“取消”按钮。其中config有以下配置项：
199       *         <ul style="margin-left:40px">
200       *             <li>title：弹出窗口的标题文字，String类型，可以使用普通字符串，也可以使用html代码。默认值为'请输入'。</li>
201       *             <li>content：弹出窗口的提示内容，String类型，可以使用普通字符串，也可以使用html代码。无默认值。</li>
202       *             <li>onClose：弹出窗口关闭时的回调函数，Function类型，点击"确定"按钮来关闭弹出窗口时，Function的参数value值为用户在输入框里输入的字符串（一定是字符串），点击“取消”按钮或按ESC键关闭弹出窗口时，Function的参数value值为false。无默认值。<b>注意：在此方法中返回false将会阻止弹出窗口关闭。</b></li>
203       *         </ul>
204       *         <br/>使用方式如下：<br/>
205       *         <pre>
206       *             $.omMessageBox.prompt({
207       *                 title:'商品数量',
208       *                 content:'请输入你要购买的商品的数量（你的余额最多只能购买12千克）：',
209       *                 onClose:function(value){
210       *                     if(value===false){ //按了取消或ESC
211       *                         alert('取消购买');
212       *                         return;
213       *                     }
214       *                     if(value==''){
215       *                         alert('数量不能为空');
216       *                         return false; //不关闭弹出窗口
217       *                     }
218       *                     if(value-0+'' !== value){
219       *                         alert('只能输入数字');
220       *                         return false;  //不关闭弹出窗口
221       *     `               }
222       *                     if(value<0 || value>12){
223       *                         alert('请输入0-12之间的数字（可带小数）');
224       *                         return false; //不关闭弹出窗口
225       *                     }else{
226       *                         alert('开始购买'+value+'千克商品');
227       *                     }
228       *                 }
229       *             });
230       *         </pre>
231       *     </li>
232       *     <li>
233       *         <b>$.omMessageBox.waiting(config | 'close')</b><br/>
234       *         弹出一个Prompt提示，有一个输入框和“确定”和“取消”按钮。该提示窗口没有关闭按钮，也不可以按ESC关闭。如果参数是'close'时表示关闭上次弹出的Waiting提示窗口。如果是config时表示要弹出一个Waiting提示窗口，其中config有以下配置项：
235       *         <ul style="margin-left:40px">
236       *             <li>title：弹出窗口的标题文字，String类型，可以使用普通字符串，也可以使用html代码。默认值为'请稍候'。</li>
237       *             <li>content：弹出窗口的提示内容，String类型，可以使用普通字符串，也可以使用html代码。无默认值。</li>
238       *         </ul>
239       *         <br/>使用方式如下：<br/>
240       *         <pre>
241       *             //弹出提示
242       *             $.omMessageBox.waiting({
243       *                 title:'请稍候',
244       *                 content:'服务器正在处理您的请求，请稍候...',
245       *             });
246       * 
247       *             //关闭提示
248       *             $.omMessageBox.waiting('close');
249       *         </pre>
250       *     </li>
251       * </ul>
252       */
253     $.omMessageBox = {
254         alert: function(config){
255             config = config || {};
256             config.title = config.title || '提示';
257             config.type = config.type || 'alert';
258             _show(config);
259         },
260         confirm: function(config){
261             config = config || {};
262             config.title = config.title || '确认';
263             config.type = 'confirm';
264             _show(config);
265         },
266         prompt: function(config){
267             config = config || {};
268             config.title = config.title || '请输入';
269             config.type = 'prompt';
270             _show(config);
271         },
272         waiting: function(config){
273             if (config === 'close') {
274                 $('.om-messageBox-waiting').remove();
275                 return;
276             }
277             config = config || {};
278             config.title = config.title || '请等待';
279             config.type = 'waiting';
280             _show(config);
281         }
282     };
283 }(jQuery));
