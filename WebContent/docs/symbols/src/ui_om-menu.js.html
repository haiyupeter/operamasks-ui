 15     /**
 16      * @name omMenu
 17      * @class menu组件。menu组件支持三种数据组织方式，分别为页面dom元素、json数据、url取值。<br/>
 18      *        基本的json格式为{id:"",label:"",icon:"img/abc.png",seperator:true,disabled:true,children:[{},{}]},其中的id和label必须设置，<br/>
 19      *        seperator为分割线，如果设置为true，则会在当前menuItem下面增加一条分割线。<br/><br/>
 20      *        组件默认会处理的属性为上面列举的6个，即id、label、icon、seperator、disabled、children。<br/>
 21      *        如果你自定义了某个属性，比如url，系统不会处理，它是在点击的时候交给onSelect方法处理，通过item.url获取url参数。
 22      * <b>特点：</b><br/>
 23      * <ol>
 24      *      <li>支持icon自定制</li>
 25      *      <li>灵活的事件处理机制，自由增加json属性，事件执行时获取数据进行处理</li>
 26      *      <li>支持动态改变menuItem的disabled属性</li>
 27      *      <li>支持右键菜单，无需指定位置，自动定位</li>
 28      *      <li>支持菜单分组，使用showSeparator属性配置</li>
 29      * </ol>
 30      * 
 31      * <b>示例：</b><br/>
 32      * <pre>
 33      * <script type="text/javascript" >
 34      * $(document).ready(function() {
 35      *      //menu定义
 36      *      $('#contextMenu').omMenu({
 37      *          contextMenu : true,
 38      *          dataSource : '../../omMenu.json'
 39      *      });
 40      *      //显示menu菜单
 41      *      $('#contextMenu_test').bind('contextmenu',function(e){
 42      *           $('#contextMenu').omMenu('show',e);
 43      *      });
 44      * });

118         show : function(triggerEle){
119             var self = this, options = self.options , top , left, element = self.element;
120             var offSet = $(triggerEle).offset();
121             if( options.contextMenu ){
122                 top = triggerEle.pageY;
123                 left = triggerEle.pageX;
124                 triggerEle.preventDefault();
125                 triggerEle.stopPropagation();
126                 triggerEle.cancelBubble=true; //IE
127             }else{
128                 var buttomWidth = parseInt($(triggerEle).css('borderBottomWidth').replace('px',''));
129                 top = offSet.top +  $(triggerEle).height() + (buttomWidth != 'NaN'?buttomWidth:0) + 1; //1px作为调节距离
130                 left = offSet.left +  1;
131             }
132             var parent = element.parent();
133             while(parent.css('position') == 'static' && parent[0].nodeName != 'BODY'){
134                 parent = parent.parent();
135             }
136             top -= parent.offset().top;
137             left -=parent.offset().left;
138             
139             if((left + element.outerWidth()) > document.documentElement.clientWidth){ //当右边距离过短的时候会将提示框调整到左边
140                 left = left - element.outerWidth() - 20;
141             }
142             $(element).css({"top":top,'left':left}).show();
143             $(element).children("ul.om-menu").show();
144             var width = $(element).width()*0.7;
145             $(element).children("ul.om-menu").children().each(function(index,li){
146                 if($(li).find("span:first").hasClass('om-menu-item-sep')){   //TODO hasClass直接导致IE9使用MaxWidth
147                     $(li).find("span:first").width('98%'); //分隔条宽度
148                 }else{
149                     if($(li).find("span:first").width() > width){
150                         $(li).find("span:first").width($(li).attr('aria-haspopup')?width-15:width); //去掉icon的padding
151                     }
152                 }
153             });

257         _appendNodes : function(source,index){
258             var self = this , menuHtml = [];
259             var ulClass = (index == undefined)?"om-menu":"om-menu-content";
260             var display = (index == undefined)?"block":"none";
261             var imgClass = (index == undefined)?"om-menu-icon" : "om-menu-icon om-menu-icon-child";
262             menuHtml.push("<ul class=\""+ulClass+" om-corner-all\" style=\"display:"+display+";\">");
263             var childrenHtml = [];
264             $(source).each(function(index , item){
265                     if(item.children != null){
266                         if(item.disabled === true || item.disabled == "true"){
267                             childrenHtml.push("<li id=\""+item.id+"\" aria-haspopup=\"true\"  class=\"om-state-disabled\">");
268                         }else{
269                             childrenHtml.push("<li id=\""+item.id+"\"  aria-haspopup=\"true\">");
270                         }
271                         childrenHtml.push("<a href=\"javascript:void(0)\" class=\"om-corner-all om-menu-indicator\">");
272                         item.icon?childrenHtml.push("<img class=\""+imgClass+"\" src=\""+item.icon+"\">"):null;
273                         item.icon?childrenHtml.push("<span>"+item.label+"</span>"):childrenHtml.push("<span style=\"margin-left:2em;\">"+item.label+"</span>");
274                         childrenHtml.push("<span class=\"ui-icon-span\" role=\"popup\"></span>");
275                         childrenHtml.push("</a>");
276                         childrenHtml.push(self._appendNodes(item.children,index++));
277                         childrenHtml.push("</li>");
278                     }else{
279                         if(item.disabled === true || item.disabled == "true"){
280                             childrenHtml.push("<li id=\""+item.id+"\"  class=\"om-state-disabled\">");
281                         }else{
282                             childrenHtml.push("<li id=\""+item.id+"\" >");
283                         }
284                         childrenHtml.push("<a href=\"javascript:void(0)\" class=\"om-corner-all om-menu-indicator\">");
285                         item.icon?childrenHtml.push("<img class=\""+imgClass+"\" src=\""+item.icon+"\">"):null;
286                         item.icon?childrenHtml.push("<span>"+item.label+"</span>"):childrenHtml.push("<span style=\"margin-left:2em;\">"+item.label+"</span>");
287                         childrenHtml.push("</a>");
288                         childrenHtml.push("</li>");
289                     }
290                     if(item.seperator == "true" || item.seperator == true){
291                         childrenHtml.push("<li class=\"om-menu-sep-li\"  ><span class=\"om-menu-item-sep\"> </span></li>");
292                     }
293                     var li = $(self.element).attr('id') + "_"+item.id;
294                     $(self.element).data(li , item);
295             });
296             menuHtml.push(childrenHtml.join(""));
297             menuHtml.push("</ul>");
298             return menuHtml.join("");

302         _parseDomMenu : function(element){
303             if(element.parent().attr("aria-haspopup") == "true"){ //判断是否为第一帧
304                 element.addClass("om-menu-content om-corner-all");
305             }
306             element.css('display','none');
307             var lis = element.children();
308             for(var i=0;i<lis.length;i++){
309                 var li = $(lis[i]) , liCul = li.children("ul");
310                 if(liCul.length > 0){
311                     li.attr("aria-haspopup","true");
312                     li.find("span[role='popup']").addClass("ui-icon-span");
313                     this._parseDomMenu(liCul);
314                 }
315                 li.find("a").addClass("om-corner-all om-menu-indicator");
316                 li.find("img").addClass("om-menu-icon");
317             }

321             if(li && li.length > 0){

323                 li_child_ul.css({"minWidth":this.options.minWidth, "top":li.position().top });

325                 if((2*left + li.offset().left) > document.documentElement.clientWidth){ //当右边距离过短的时候会将提示框调整到左边

327                 }

329                 li_child_ul.css("left",left);
330                 li_child_ul.show();
331                 
332                 li_child_ul.children().each(function(index,li){
333                     if($(li).find("span:first").hasClass('om-menu-item-sep')){

336                         if(li_child_ul.width() > self.options.maxWidth){
337                             li_child_ul.width(self.options.maxWidth);
338                             width = self.options.maxWidth * 0.6;
339                         }else{
340                             li_child_ul.width(li_child_ul.width() + 10);

342                         }
343                         $(li).find("span:first").width(width);

383         _bindEvent : function(){
384             var self = this , element = self.element,
385             	uls = element.find("ul"),
386             	lis = element.find("li"),
387             	$doc = $(document),
388             	tempEvent;
389             for(var i=0 ; i<lis.length ; i++){
390                 if(!$(lis[i]).hasClass("om-state-disabled")){
391                    self._bindLiEvent(lis[i]);
392                 }
393             };
394             for(var j=0 ; j<uls.length ; j++){
395                 $(uls[j]).bind("mouseleave.menuContainer",function(){
396                     var ul = $(this);
397                     if(ul.parent().attr("aria-haspopup") == "true"){
398                         ul.hide();
399                     }
400                 });
401             };
402             this.globalEvent = [];
403             $doc.bind('mousedown.omMenu',tempEvent=function(){
404                 self._hide();
405             });
406             this.globalEvent.push(tempEvent);
407             $doc.bind('keyup.omMenu' , tempEvent=function(e){
408                 var key = e.keyCode,
409                     keyEnum = $.om.keyCode;
410                 switch (key) {
411                 case keyEnum.DOWN: //down
412                     self._selectNext();
413                     break;
414                 case keyEnum.UP: //up
415                     self._selectPrev();
416                     break;
417                 case keyEnum.LEFT: //left
418                     self._hideRight();
419                     break;
420                 case keyEnum.RIGHT: //right
421                     self._showRight();
422                     break;
423                 case keyEnum.ENTER: //enter 建立在当前menu就是打开的menu前提下
424                     if(element.css("display") == "block")
425                         self._backfill(element);
426                     break;
427                 case keyEnum.ESCAPE: //esc
428                     self._hide();
429                     break;
430                 default:
431                    null;
432                 }
433             });
434             this.globalEvent.push(tempEvent);
435             $doc.bind('keydown.omMenu' , tempEvent=function(e){//fixed AOM-430
436             	if(e.keyCode >= 37 && e.keyCode <= 40){
437             		e.preventDefault();
438             	}
439             });
440             this.globalEvent.push(tempEvent);
441         },

451         /**
452          * 排除掉分隔条的干扰，找到下一个menuItem

456         _findNext : function(liMenuItem){
457             var next,
458                 oldItem = liMenuItem;
459             while( (next=liMenuItem.next("li")).length !== 0 ){
460             	if(!next.hasClass("om-menu-sep-li") && !next.hasClass("om-state-disabled")){
461             		return next;
462             	}
463             	liMenuItem = next;
464             }
465             //如果没有，则从上到下再找一遍
466             var item = oldItem.parent().find("li:first");
467             while(item.length !== 0 && item != oldItem){
468             	if(!item.hasClass("om-menu-sep-li") && !item.hasClass("om-state-disabled")){
469             		return item;
470             	}
471             	item = item.next("li");
472             }
473         },
474         /**
475          * 排除掉分隔条的干扰，找到前一个menuItem
476          * 返回li
477          * @param liMenuItem
478          */
479         _findPrev : function(liMenuItem){
480             var prev,
481             	oldItem = liMenuItem;
482             while( (prev=liMenuItem.prev("li")).length !== 0 ){
483             	if(!prev.hasClass("om-menu-sep-li") && !prev.hasClass("om-state-disabled")){
484             		return prev;
485             	}
486             	liMenuItem = prev;
487             }
488             //如果没有，则从下往上再找一遍
489             var ulChildren = oldItem.parent().children();
490             var item =ulChildren.eq(ulChildren.length - 1);
491             while(item.length !== 0 && item != oldItem){
492             	if(!item.hasClass("om-menu-sep-li") && !item.hasClass("om-state-disabled")){
493             		return item;
494             	}
495             	item = item.prev("li");
496             }
497         },
498         _selectNext : function(){
499             var self = this , element = self.element ,curLi;
500             var menuItemHover = element.find("li.om-menu-item-hover");
501             var hoverLast = menuItemHover.eq(menuItemHover.length-1);
502             if(menuItemHover.length == 0){ //如果没有被选中的就选中第一个
503                 curLi = element.find("li").eq(0);
504                 while(curLi.hasClass('om-state-disabled')){
505                     curLi = curLi.next('li');
506                 }
507                 curLi.addClass("om-menu-item-hover");
508             }else{
509                 curLi = self._findNext(hoverLast);
510                 if(curLi.length <= 0) return;
511                 curLi.addClass("om-menu-item-hover");
512                 hoverLast.removeClass("om-menu-item-hover");
513             }
514             this._hideChildren(hoverLast);
515             this._showChildren(curLi);
516         },
517         _selectPrev : function(){
518             var self = this , element = self.element,curLi;
519             var menuItemHover = element.find("li.om-menu-item-hover");
520             var hoverLast = menuItemHover.eq(menuItemHover.length-1);
521                 curLi = element.find("ul.om-menu > li");
522             if(menuItemHover.length == 0){ //如果没有被选中的就选中最后一个
523                 var lastLi = curLi.eq(curLi.length-1) , i=1;
524                 while(lastLi.hasClass('om-state-disabled')){
525                     lastLi = curLi.eq(curLi.length-(i++));
526                 }
527                 (curLi = lastLi).addClass("om-menu-item-hover");
528             }else{
529                 curLi = self._findPrev(hoverLast);
530                 if(curLi.length <= 0) return;
531                 curLi.addClass("om-menu-item-hover");
532                 hoverLast.removeClass("om-menu-item-hover");
533             }
534             this._hideChildren(hoverLast);
535             this._showChildren(curLi);
536         },
537         _hideRight : function(){
538             var self = this , element = self.element;
539             var currentA = element.find("li.om-menu-item-hover") , 
540                 hoverLast = currentA.eq(currentA.length - 1);
541             hoverLast.removeClass("om-menu-item-hover");
542             self._hideChildren(hoverLast);
543         },
544         _showRight : function(){
545             var self = this , element = self.element,curLi;
546             var parentA = element.find("li.om-menu-item-hover") , 
547                 parentLi = parentA.eq(parentA.length - 1);
548             if(parentLi.attr("aria-haspopup") == "true"){
549                 curLi = parentLi.children("ul").find("li").eq(0);
550                 curLi.addClass("om-menu-item-hover");
551             }
552             self._showChildren(curLi);
553         },
554         _backfill : function(element){
555             var curas = element.find("li.om-menu-item-hover");
556             curas.eq(curas.length - 1).mousedown();
557         }
