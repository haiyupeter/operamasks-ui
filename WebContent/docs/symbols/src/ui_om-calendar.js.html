503         /**

506          * @private

508         _templetShow: function(templet, data) {
509             var str_in,value_s,i,m,value,par;
510             if (data instanceof Array) {
511                 str_in = '';
512                 for (i = 0; i < data.length; i++) {
513                     str_in += arguments.callee(templet, data[i]);
514                 }
515                 templet = str_in;
516             } else {
517                 value_s = templet.match(/{\$(.*?)}/g);
518                 if (data !== undefined && value_s !== null) {
519                     for (i = 0,m = value_s.length; i < m; i++) {
520                         par = value_s[i].replace(/({\$)|}/g, '');
521                         value = (data[par] !== undefined) ? data[par] : '';
522                         templet = templet.replace(value_s[i], value);
523                     }
524                 }
525             }
526             return templet;
527         },
