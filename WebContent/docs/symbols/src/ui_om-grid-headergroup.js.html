 26 		$.extend(this , {

 32 					op = this.options,
 33 					$hDiv = this.hDiv,
 34 					$ths = $hDiv.find("th[axis^='col']");

 36 					
 37 					result.sort(function(first , second){
 38 						return first.axis.slice(3) - second.axis.slice(3);
 39 					});

 41 					//op.rowDetailsProvider && result.unshift($hDiv.find("th[axis='expenderCol']")[0]);
 42 					!op.singleSelect && result.unshift($hDiv.find("th[axis='checkboxCol']")[0]);

 48 				var op=this.options,
 49 	                $elem=this.element,
 50 	                $grid = $elem.closest('.om-grid'),
 51 	                cm= op.colModel,
 52 	                allColsWidth = 0, //colModel的宽度
 53 	                autoExpandColIndex = -1,
 54 	                tmp = "<th class='$' $ $ $ $ $><div class='$' style='text-align:$; $'>$</div></th>",
 55 	                content = ["<thead>"],

 57 	                item,

139 	function _getColIndex(self , header){
140 		var cm = self._getColModel();
141 		for(var i=0,len=cm.length; i<len; i++){
142 			if(cm[i].name == header.name){
143 				return i;
144 			}
145 		}
146 	}
147 	
148 	function _buildTh(content , cols , tmp){
149     	var j=0;
150 		content.push(tmp.replace(/\$/g , function(){
151 			return cols[j++];
152 		}));
153 	}

155 	/**
156 	 * 创建只包含最基本数据的colModel。也就是创建出没有多表头时的基本colModel。注意，生成的colModel是具有顺序的。
157 	 */
158 	function _buildBasicColModel(self){
159 		//基本colModel缓存引用
160 		self._colModel = [];
161 		var cm = self._getColModel(),
162 			matrix = [],
163 			realRowHeader = [],//colModel中非最后一行的具有真实数据意义的header缓存
164 			rows = 1,
165 			cols = 1,
166 			rowHeader,
167 			item,
168 			colIndex,//每个header从哪一列开始渲染
169 			len = cm.length;
170 
171 		for(var i=0; i<len; i++){
172 			matrix[i] = [];
173 		}
174 		for(var i=0,row=len; i<row; i++){
175 			rowHeader = cm[i];
176 			
177 			for(var j=0,col=rowHeader.length; j<col; j++){
178 				item = rowHeader[j];
179 				rows = item.rowspan || 1;
180 				cols = item.colspan || 1;
181 				
182 				colIndex = _checkMatrix(matrix , i , rows , cols);
183 				if(item.name){
184 					realRowHeader.push({header:item , colIndex:colIndex});
185 				}
186 			}
187 		}
188 		realRowHeader.sort(function(first , second){
189 			return first.colIndex > second.colIndex;
190 		});
191 		i=0;
192 		while(realRowHeader[i] && self._colModel.push(realRowHeader[i++].header));
193 	}
194 	
195 	function _checkMatrix(matrix , index , rows , cols){
196 		var i=0;
197 		while(matrix[index][i] && ++i);
198 		for(var j=index; j<index+rows; j++){
199 			for(var k=i; k<i+cols; k++){
200 				matrix[j][k] = true;
201 			}
202 		}
203 		return i;
204 	}
