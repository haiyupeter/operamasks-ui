 19      * <pre>
 20      * {

389         // handle toggle event
390         // change the target to the treenode (li dom)
391         toggler: function(target) {
392             var self = this,
393                 options = self.options;
394             var nid = target.attr("id");
395             var node = self.findByNId(nid);
396             var hidden = target.hasClass(CLASSES.expandable);
397             
398             if ( hidden ) {
399                 var onBeforeExpand = options.onBeforeExpand;
400                 if(onBeforeExpand && false === self._trigger("onBeforeExpand",null,node)){
401                     return self;
402                 }
403             } else {
404                 var onBeforeCollapse = options.onBeforeCollapse;
405                 if(onBeforeCollapse && false === self._trigger("onBeforeCollapse",null,node)){
406                     return self;
407                 }
408             }
409             
410             // swap classes for hitarea
411             var hitarea = target.find( target.find(">.hitarea") );
412             self._swapClass(hitarea, CLASSES.collapsableHitarea, CLASSES.expandableHitarea);
413             self._swapClass(hitarea, CLASSES.lastCollapsableHitarea, CLASSES.lastExpandableHitarea);
414             
415             // swap classes for li
416             self._swapClass(target, CLASSES.collapsable, CLASSES.expandable);
417             self._swapClass(target, CLASSES.lastCollapsable, CLASSES.lastExpandable);
418             
419             // find child lists
420             target.find( ">ul" )
421                 .each(function(){
422                     if ( hidden ) {
423                         $(this).show();
424                         var onExpand = options.onExpand;
425                         onExpand && self._trigger("onExpand",null,node);
426                     } else {
427                         $(this).hide();
428                         var onCollapse = options.onCollapse;
429                         onCollapse && self._trigger("onCollapse",null,node);
430                     }
431                 });
432         },

435             var self = this, options = self.options,
436                 target = self.element,
437                 source = options.dataSource;
438             target.empty();

440                 if(typeof source == 'string'){

444                 		source = self._transformToNodes(source);
445                 	}
446                 	target.data("init_dataSource", source);
447                     self._appendNodes.apply(self, [target, source]);
448                     self.updateNode(target);
449                 }
450             }

455                 options = this.options,

458                 onError = options.onError;

476                 }

594         select: function(target) {
595             var self = this,
596                 options = this.options,
597                 onBeforeSelect = options.onBeforeSelect,
598                 onSelect = options.onSelect;
599             if(onBeforeSelect && false === self._trigger("onBeforeSelect",null,target)) {

601             }
602             var node = $("#" + target.nid);
