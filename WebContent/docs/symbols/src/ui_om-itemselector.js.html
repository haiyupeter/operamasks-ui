272         _create:function(){

279         _init:function(){
280             var op=this.options,
281                 dataSource=op.dataSource;
282             $('>legend>span',this.leftPanel).html($.om.lang._get(op,"omItemSelector","availableTitle"));

285             this._buildToolbar();

289                 var self=this;

291                     url: dataSource,
292                     method: 'GET',
293                     dataType: 'json',
294                     success: function(data, textStatus){
295                         if (self._trigger("onSuccess",null,data,textStatus) === false) {
296                             return;
297                         }
298                         data=op.preProcess(data);
299                         op.dataSource=data;
300                         self._buildList();
301                     },
302                     error: function(XMLHttpRequest, textStatus, errorThrown){
303                     	self._trigger("onError",null,XMLHttpRequest, textStatus, errorThrown);
304                     }
305                 });
306             }else{
307                 this._buildList();
308             }

340         _buildList:function(){
341             var op=this.options;
342                 dataSource = op.dataSource,
343                 value = op.value,
344                 fmt=op.clientFormatter,
345                 leftHtml='',
346                 rightHtml='',
347                 // {text:'abc',value:2}的value是否在value:[0,2,4]这样的数组中
348                 inArray=function(data,valueArr){
349                     for(var i=0,len=valueArr.length;i<len;i++){
350                         if(data.value===valueArr[i]){
351                             return true;
352                         }
353                     }
354                     return false;
355                 },
356                 buildHtml=fmt?function(index,data){
357                     return '<dt _index="'+index+'">'+fmt(data,index)+'</dt>';
358                 }:function(index,data){
359                     return '<dt _index="'+index+'">'+data.text+'</dt>';
360                 };
361             if($.isArray(dataSource) && jQuery.isArray(value)){
362                 $.each(dataSource,function(index,data){
363                     if(inArray(data,value)){//在value中，要放到右边
364                         rightHtml+=buildHtml(index,data);
365                     }else{//不在value中,放到左边
366                         leftHtml+=buildHtml(index,data);
367                     }
368                 });
369             }
370             $('.om-itemselector-items>dl',this.leftPanel).html(leftHtml);
371             $('.om-itemselector-items>dl',this.rightPanel).html(rightHtml);
372             this._makeDraggable();
373         },
374         _makeDraggable:function(){
375             var self=this,
376                 autoSort=this.options.autoSort,
377                 allItems=$('.om-itemselector-items>dl',this.element);
378             allItems.attr('onselectstart',"javascript:return false;");//禁止选择鼠标拖动文字
379             if(!autoSort){
380                 allItems.sortable({
381                     placeholder:'om-itemselector-placeholder',
382                     start:function(ui,event){
383                         $('.om-itemselector-items>dl>dt',self.element).removeClass('om-state-highlight');
384                         ui.item.addClass('om-state-highlight');
385                         ui.item.oldIndex = ui.item.index(); 
386                     },
387                     stop:function(ui,event){
388                     	self._trigger("onSort",null,self.options.dataSource[ui.item.attr('_index')],ui.item.oldIndex,ui.item.index());
389                     }
390                 }).disableSelection();
391             }
392         },

412             });
413             this.rightPanel.delegate('.om-itemselector-items>dl>dt','dblclick',function(){
414                 self._moveItemsToTarget('.om-state-highlight',false);
415             });
416             //右移
417             $('.om-itemselector-tbar-add',toolbar).click(function(){

419             });
420             //左移
421             $('.om-itemselector-tbar-remove',toolbar).click(function(){

427             });
428             //全部左移
429             $('.om-itemselector-tbar-removeAll',toolbar).click(function(){
430                 self._moveItemsToTarget('',false);
431             });
432             //autoSort=false时才可以排序
433             if(!op.autoSort){
434                 var selector='.om-itemselector-items>dl>dt.om-state-highlight:first';
435                 //置顶
436                 $('.om-itemselector-tbar-moveTop',toolbar).click(function(){
437                     var panel=self.rightPanel,
438                         selectedItems=$(selector,panel);
439                     if(selectedItems.size()==0){
440                         panel=self.leftPanel;
441                         selectedItems=$(selector,panel);
442                     }
443                     var oldIndex=selectedItems.index();
444                     if(selectedItems.size()==0 || oldIndex==0)
445                         return;
446                     selectedItems.prependTo($('.om-itemselector-items>dl',panel));
447                     self._trigger("onSort",null,op.dataSource[selectedItems.attr('_index')],oldIndex,0);
448                 });
449                 //置底
450                 $('.om-itemselector-tbar-moveBottom',toolbar).click(function(){
451                     var panel=self.rightPanel,
452                         selectedItems=$(selector,panel);
453                     if(selectedItems.size()==0){
454                         panel=self.leftPanel;
455                         selectedItems=$(selector,panel);
456                     }
457                     var oldIndex=selectedItems.index(),
458                         newIndex=selectedItems.siblings().size();
459                     if(selectedItems.size()==0 || oldIndex==newIndex)
460                         return;
461                     selectedItems.appendTo($('.om-itemselector-items>dl',panel));
462                     self._trigger("onSort",null,op.dataSource[selectedItems.attr('_index')],oldIndex,newIndex);
463                 });
464                 //上移
465                 $('.om-itemselector-tbar-moveUp',toolbar).click(function(){
466                     var panel=self.rightPanel,
467                         selectedItems=$(selector,panel);
468                     if(selectedItems.size()==0){
469                         panel=self.leftPanel;
470                         selectedItems=$(selector,panel);
471                     }
472                     var oldIndex=selectedItems.index();
473                     if(selectedItems.size()==0 || oldIndex==0)
474                         return;
475                     selectedItems.insertBefore(selectedItems.prev());
476                     self._trigger("onSort",null,op.dataSource[selectedItems.attr('_index')],oldIndex,oldIndex-1);
477                 });
478                 //下移
479                 $('.om-itemselector-tbar-moveDown',toolbar).click(function(){
480                     var panel=self.rightPanel,
481                         selectedItems=$(selector,panel);
482                     if(selectedItems.size()==0){
483                         panel=self.leftPanel;
484                         selectedItems=$(selector,panel);
485                     }
486                     var oldIndex=selectedItems.index();
487                     if(selectedItems.size()==0 || oldIndex==selectedItems.siblings().size())
488                         return;
489                     selectedItems.insertAfter(selectedItems.next());
490                     self._trigger("onSort",null,op.dataSource[selectedItems.attr('_index')],oldIndex,oldIndex+1);
491                 });
492             }

494         _moveItemsToTarget:function(selector,isLeftToRight){
495             var formPanel=isLeftToRight?this.leftPanel:this.rightPanel,
496                 selectedItems=$('.om-itemselector-items>dl>dt'+selector,formPanel);
497             if(selectedItems.size()==0)
498                 return;
499             var toPanel=isLeftToRight?this.rightPanel:this.leftPanel,
500                 op=this.options,
501                 itemData=[];
502             selectedItems.each(function(){
503                 itemData.push(op.dataSource[$(this).attr('_index')]);
504             });
505             //先触发onBeforeItemSelect或onBeforeItemDeselect事件，然后移动并触发onItemSelect或onItemDeselect事件
506             if(isLeftToRight){
507                 if(this._trigger("onBeforeItemSelect",null,itemData)===false){
508                     return;
509                 }
510                 selectedItems.appendTo($('.om-itemselector-items>dl',toPanel));
511                 this._trigger("onItemSelect",null,itemData);
512             }else{
513                 if(this._trigger("onBeforeItemDeselect",null,itemData)===false){
514                     return;
515                 }
516                 selectedItems.appendTo($('.om-itemselector-items>dl',toPanel));
517                 this._trigger("onItemDeselect",null,itemData);
518             }
519             //如果设了autoSort=true则自动排序
520             if(op.autoSort){
521                 var result=$('.om-itemselector-items>dl>dt',toPanel).sort(function(a,b){
522                     return $(a).attr('_index')-$(b).attr('_index');
523                 });
524                 selectedItems.parent().append(result);
525             }
526         },
527         
528         /**
529          * 得到或设置组件的value值。
530          * @function
531          * @name omItemSelector#value
532          * @param v 设置的值，不设置表示获取值
533          * @returns 如果没有参数时表示getValue()返回combo的value值，比如dataSource:[{text:'abc',value:true},{text:'def',value:2},{text:'xyz',value:'x'}]选择了第2条和第三条，则getValue返回[2,'x']。如果有参数时表示setValue(newValue)返回jQuery对象。
534          * 
535          */
536         value:function(newValue){
537             if(arguments.length==0){ //getValue
538                 var op=this.options,
539                     selectedItems=$('.om-itemselector-items>dl>dt',this.rightPanel),
540                     returnValue=[];
541                 selectedItems.each(function(){
542                     returnValue.push(op.dataSource[$(this).attr('_index')].value);
543                 });
544                 return returnValue;
545             }else{ //setValue
546                 if($.isArray(newValue)){
547                     this.options.value=newValue;
548                     this._buildList();
549                 }
550             }
551         }
