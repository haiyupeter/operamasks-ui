 17     /** 
 18      * @name omBorderLayout
 19      * @class omBroderLayout是页面布局的基础组件.把页面拆分为north,south,west,center,east（上、下、左、中、右）5个区域，除了center是必须设置的之外其他的都是可选的。<br/>
 20      * <b>特点：</b><br/>
 21      * <ol>
 22      * 		<li>以omPanel作为子组件，五个区域(north,south,west,center,east)都用omPanel实现。即每个panel都支持omPanel中的属性设置。</li>
 23      * 		<li>在omPanel的基础上添加两个属性：region表示所属区域，resizable表示区域是否可拖拉改变大小。</li>
 24      * 		<li>可设置每个区域之间的间隔大小。</li>
 25      * 		<li>可设置borderLayout自动适应父容器的大小。</li>
 26      * </ol>
 27      * <b>示例：</b><br/>
 28      * <pre>
 29      * <script type="text/javascript" >
 30      * $(document).ready(function() {
 31      *     $('#page').omBorderLayout({
 32      *     panels:[{ 
 33      *        id:"north-panel", 
 34      *        title:"This is north panel", 
 35      *        region:"north", 
 36      *        resizable:true, 
 37      *        collapsible:true 
 38      *    },{ 
 39      *        id:"center-panel", 
 40      *        title:"This is center panel", 
 41      *        region:"center" 
 42      *    },{ 
 43      *        id:"west-panel", 
 44      *        title:"This is west panel", 
 45      *        region:"west", 
 46      *        resizable:true, 
 47      *        collapsible:true, 
 48      *        width:200 
 49      *    },{ 
 50      *        id:"east-panel", 
 51      *        title:"This is east panel", 
 52      *        region:"east", 
 53      *        resizable:true, 
 54      *        collapsible:true, 
 55      *        width:100 
 56      *    }], 
 57      *    spacing:3 
 58      * });
 59      * </script>
 60      * 
 61      * <div id="page" style="width:800px;height:600px;">
 62      *	<div id="north-panel" />
 63      *	<div id="center-panel" />
 64      *	<div id="west-panel" />
 65      *	<div id="east-panel" />
 66 	 * </div>
 67 	 * </pre>
 68      * @constructor
 69      * @description 构造函数. 
 70      * @param p 标准config对象：{}
 71      */

 84 		     *        title:"This is north panel", 
 85 		     *        region:"north", 
 86 		     *        resizable:true, 
 87 		     *        collapsible:true 
 88 		     *    },{ 
 89 		     *        id:"center-panel", 
 90 		     *        title:"This is center panel", 
 91 		     *        region:"center" 
 92 		     *    },{ 
 93 		     *        id:"west-panel",

136 		_create : function() {
137 			if(!this.options.panels) return;
138 			// 设置region拖拉改变宽度的最小值
139 			this._minWidth = 50;
140 			// 设置region拖拉改变高度的最小值
141 			this._minHeight = 28;
142 			this._buildRegion();
143 			this._resizeRegion();
144 			$(window).resize($.proxy(this, "_resizeRegion"));
145 		},
146 		// 获取区域的大小，如果区域被隐藏了则获取代理区域(regionProxy)的大小，如果代理区域也被隐藏则返回0
147 		_getRegionSize : function(region){
148 			var $region = this._getRegion(region),
149 				$proxy = this._getRegionProxy(region),
150 				size = {};
151 			size.width = this._regionVisible($region)?$region.outerWidth(true):
152 				(this._regionVisible($proxy)?$proxy.outerWidth(true):0);
153 			size.height = this._regionVisible($region)?$region.outerHeight(true):
154 				(this._regionVisible($proxy)?$proxy.outerHeight(true):0);
155 			return size;
156 		},
157 		_resizeRegion : function() {
158 			var $centerRegion = this._getRegion("center"),
159 				$northRegion = this._getRegion("north"),
160 				$southRegion = this._getRegion("south"),
161 				$westRegion = this._getRegion("west"),
162 				$eastRegion = this._getRegion("east"),
163 
164 				$northProxy = this._getRegionProxy("north"),
165 				$southProxy = this._getRegionProxy("south"),
166 				$westProxy = this._getRegionProxy("west"),
167 				$eastProxy = this._getRegionProxy("east"),
168 				
169 				northHeight = this._getRegionSize("north").height;
170 				southHeight = this._getRegionSize("south").height;
171 				westWidth = this._getRegionSize("west").width;
172 				eastWidth = this._getRegionSize("east").width;
173 				layoutWidth = this.element.width();
174 				layoutHeight = this.element.height();
175 			
176 			$northProxy && $northProxy.outerWidth(layoutWidth);
177 			$northRegion && $northRegion.find(">.om-panel-body").omPanel("resize",{width:layoutWidth});
178 
179 			$southProxy && $southProxy.outerWidth(layoutWidth);
180 			$southRegion && $southRegion.find(">.om-panel-body").omPanel("resize",{width:layoutWidth});
181 				
182 			$centerRegion.css({top:northHeight,left:westWidth});
183 			$centerRegion.find(">.om-panel-body").omPanel("resize",{
184 				width:layoutWidth - westWidth - eastWidth,
185 				height:layoutHeight - northHeight - southHeight
186 			});
187 			var centerHeight = $centerRegion.outerHeight(true);
188 			if($southRegion){
189 				$southRegion.css({top:layoutHeight-$southRegion.outerHeight(true)});
190 			}
191 			if($westRegion){
192 				$westRegion.css({top:northHeight});
193 				$westRegion.find(">.om-panel-body").omPanel("resize",{height:centerHeight});
194 				if($westProxy){
195 					$westProxy.css({top:northHeight});
196 					$westProxy.outerHeight(centerHeight);
197 				}
198 			}
199 			if($eastRegion){
200 				$eastRegion.css({top:northHeight});
201 				$eastRegion.find(">.om-panel-body").omPanel("resize",{height:centerHeight});
202 				if($eastProxy){
203 					$eastProxy.css({top:northHeight});
204 					$eastProxy.outerHeight(centerHeight);
205 				}
206 			}
207 			
208 		},
209 		_regionVisible : function($region){
210 			return $region && $region.css("display") != "none";
211 		},
212 		_createRegionProxy : function(panel,showCollapsTrigger){
213 			var _self = this;
214 			var proxyHtml = "";
215 			if(showCollapsTrigger){
216 				proxyHtml = "<div class=\"om-borderlayout-proxy om-borderlayout-trigger-proxy-"+panel.region+"\" proxy=\""+panel.region+"\">" +
217 							"<div class=\"om-borderlayout-expand-trigger\">"+
218 							"</div>"+
219 							"</div>";
220 				var $proxy = $(proxyHtml);
221 				if(panel.region == "west" || panel.region == "east"){
222 					$proxy.width(_self.options.spacing);
223 				} else if(panel.region == "north" || panel.region == "south"){
224 					$proxy.height(_self.options.spacing);
225 				}
226 				(function(panel){
227 					$proxy.find(".om-borderlayout-expand-trigger").click(function(){
228 						_self.expandRegion(panel.region);
229 					});
230 				})(panel);
231 			} else{
232 				proxyHtml = "<div class=\"om-borderlayout-proxy om-borderlayout-proxy-"+panel.region+"\" proxy=\""+panel.region+"\">" +
233 							"<div class=\"om-panel-title\"></div>"+
234 							"<div class=\"om-panel-tool\">"+
235 							"<div class=\"om-icon panel-tool-expand\">"+
236 							"</div>"+
237 							"</div>"+
238 							"</div>";
239 				var $proxy = $(proxyHtml);
240 				(function(panel){
241 					$proxy.find(".panel-tool-expand").hover(function(){
242 						$(this).toggleClass("panel-tool-expand-hover");
243 					}).click(function(){
244 						_self.expandRegion(panel.region);
245 					});
246 				})(panel);
247 			}
248 			$proxy.hover(function(){
249 				$(this).toggleClass("om-borderlayout-proxy-hover");
250 			}).appendTo(this.element);
251 		},
252 		// 构建布局框架
253 		_buildRegion : function() {
254 			var _self = this;
255 			var $layout = this.element;
256 			this.element.addClass("om-borderlayout");
257 			if (this.options.fit) {
258 				$layout.css({
259 					"width" : "100%",
260 					"height" : "100%"
261 				});
262 			}
263 			for ( var i = 0; i < this.options.panels.length; i++) {
264 				var panel = $.extend({},this.options.panels[i]);
265 				var $panelEl = this.element.find("#" + panel.id);
266 				// 是否在panel之间的间隔上显示收缩/展开panel的trigger按钮
267 				var showCollapsTrigger = panel.collapsible && _self.options.hideCollapsBtn;
268 				// 添加代理工具条
269 				if(panel.collapsible && panel.region != "center"){
270 					this._createRegionProxy(panel,showCollapsTrigger);
271 				}
272 				
273 				// 扩展panel初始化参数，添加一些必要的事件
274 				if(panel.collapsible){
275 					$.extend(panel,{
276 						collapsible:false
277 					});
278 					if(!_self.options.hideCollapsBtn){
279 						$.extend(panel,{
280 							tools:[{
281 								iconCls:["panel-tool-collapse","panel-tool-collapse-hover"],
282 								handler:function(widget){
283 									_self.collapseRegion(widget.element.parent().attr("region"));
284 								}
285 							}]
286 						});
287 					}
288 				}
289 				if(panel.closable){
290 					var oldPanelOnClose = panel.onClose;
291 					$.extend(panel,{
292 						onClose:function(){
293 							oldPanelOnClose && oldPanelOnClose.call($panelEl[0]);
294 							_self._resizeRegion();
295 						}
296 					});
297 				}
298 				
299 				
300 				// 构建panel组件
301 				$panelEl.omPanel(panel);
302 				
303 				// 初始化north和south的宽度
304 				if(panel.region == "north" || panel.region == "south"){
305 					$panelEl.omPanel("resize",{"width":$layout.width()});
306 				}
307 				
308 				var margin = "0",
309 					spacing = this.options.spacing + "px";
310 				// 给panel添加resize功能
311 				if(panel.resizable && panel.region != "center"){
312 					var handles = "";
313 						handleClass = {};
314 					if(panel.region == "west"){
315 						handles = "e";
316 						handleClass.width = spacing;
317 						handleClass.right = "-" + spacing;
318 					} else if(panel.region == "east"){
319 						handles = "w";
320 						handleClass.width = spacing;
321 						handleClass.left = "-" + spacing;
322 					} else if(panel.region == "south"){
323 						handles = "n";
324 						handleClass.height = spacing;
325 						handleClass.top = "-" + spacing;
326 					} else if(panel.region == "north"){
327 						handles = "s";
328 						handleClass.height = spacing;
329 						handleClass.bottom = "-" + spacing;
330 					}
331 					$panelEl.parent().omResizable({
332 						handles : handles,
333 						helper : "om-borderlayout-resizable-helper-" + handles,
334 						stop : function(ui,event){
335 							$layout.find(">.om-borderlayout-mask").remove();
336 							ui.element.find(">.om-panel-body").omPanel("resize",ui.size);
337 							_self._resizeRegion();
338 						},
339 						start : function(ui,event){
340 							var helper = ui.element.omResizable("option","helper");
341 							// 修改resizable的helper的宽/高为spacing大小
342 							$("body").find("." + helper).css("border-width",_self.options.spacing);
343 							// 限制拖拉改变大小的范围
344 							var region = ui.element.attr("region"),
345 								maxWidth = $layout.width() - 2*_self._minWidth,
346 								maxHeight = $layout.height() - 2*_self._minHeight;
347 							if(region == "west"){
348 								maxWidth = $layout.width() - (_self._getRegionSize("east").width + _self._minWidth);
349 								ui.element.omResizable( "option", "maxWidth", maxWidth );
350 							} else if(region == "east"){
351 								maxWidth = $layout.width() - (_self._getRegionSize("west").width + _self._minWidth);
352 								ui.element.omResizable( "option", "maxWidth", maxWidth );
353 							} else if(region == "north"){
354 								maxHeight = $layout.height() - (_self._getRegionSize("south").height + _self._minHeight + _self.options.spacing);
355 								ui.element.omResizable( "option", "maxHeight", maxHeight );
356 							} else if(region == "south"){
357 								maxHeight = $layout.height() - (_self._getRegionSize("north").height + _self._minHeight + _self.options.spacing);
358 								ui.element.omResizable( "option", "maxHeight", maxHeight );
359 							}
360 							$('<div class="om-borderlayout-mask"></div>').css({
361 								width:$layout.width(),
362 								height:$layout.height()
363 							}).appendTo($layout);
364 						},
365 						minWidth : _self._minWidth,
366 						minHeight : _self._minHeight
367 						
368 					});
369 					$panelEl.parent().find(".om-resizable-handle").css(handleClass);
370 					margin = (panel.region == "south" ? spacing : 0) + " " +
371 							 (panel.region == "west" ? spacing : 0) + " " +
372 							 (panel.region == "north" ? spacing : 0) + " " +
373 							 (panel.region == "east" ? spacing : 0);
374 					
375 					// 如果隐藏收缩panel的按钮，则在panel之间的间隔条上显示收缩/展开trigger按钮
376 					if(showCollapsTrigger){
377 						var $collapsTrigger = $("<div class='om-borderlayout-collaps-trigger-"+panel.region+"'></div>");
378 						(function($panel){$collapsTrigger.click(function(){
379 							_self.collapseRegion($panel.attr("region"));
380 						});})($panelEl.parent());
381 						$panelEl.parent().find(".om-resizable-handle").append($collapsTrigger);
382 					}
383 				}
384 				
385 				$panelEl.parent()
386 					   .addClass("om-borderlayout-region")
387 					   .addClass("om-borderlayout-region-" + panel.region)
388 					   .css("margin",margin)
389 					   .attr("region",panel.region);
390 				//添加header class用来区别borderlayout和borderlayout中内嵌的panel使用的tools 图片
391 				$panelEl.prev().addClass("om-borderlayout-region-header");
392 			}
393 		},
394 		_getRegion : function(region){

397 		},
398 		_getRegionProxy : function(region){

401 		},
402 		_getPanelOpts : function(region){
403 			for(var i = 0; i < this.options.panels.length; i++){
404 				if(region == this.options.panels[i].region){
405 					return this.options.panels[i];
406 				}
407 			}

409 		},
410         /**
411          * 折叠某个区域的panel。
412          * @name omBorderLayout#collapseRegion
413          * @function
414          * @param region 区域名称
415          * @example
416          * //折叠north区域的panel
417          * $('#page').omBorderLayout('collapseRegion', 'north');
418          */
419 		collapseRegion : function(region){
420 			var panel = this._getPanelOpts(region);
421 			if(!panel || !panel.collapsible){
422 				return;
423 			}
424 			var $region = this._getRegion(region);
425 				$body = $region.find(">.om-panel-body");
426 			if($region){
427 				var panelInstance = $.data($body[0],"omPanel");
428 				if(panelInstance.options.closed) return;
429 				if(panel.onBeforeCollapse && panelInstance._trigger("onBeforeCollapse") === false){
430 					return false;
431 				}
432 				$region.hide();
433 				panel.onCollapse && panelInstance._trigger("onCollapse");
434 				this._getRegionProxy(region).show();
435 				this._resizeRegion();
436 			}
437 		},
438 		/**
439 		 * 展开某个区域的panel。
440 		 * @name omBorderLayout#expandRegion
441 		 * @function
442 		 * @param region 区域名称
443 		 * @example
444 		 * //展开north区域的panel
445 		 * $('#page').omBorderLayout('expandRegion', 'north');
446 		 */
447 		expandRegion : function(region){
448 			var panel = this._getPanelOpts(region);
449 			if(!panel || !panel.collapsible){
450 				return;
451 			}
452 			var $region = this._getRegion(region);
453 				$body = $region.find(">.om-panel-body");
454 			if($region){
455 				var panelInstance = $.data($body[0],"omPanel");
456 				if(panelInstance.options.closed) return;
457 				if(panel.onBeforeExpand && panelInstance._trigger("onBeforeExpand") === false){
458 					return false;
459 				}
460 				$region.show();
461 				panel.onExpand && panelInstance._trigger("onExpand");
462 				this._getRegionProxy(region).hide();
463 				this._resizeRegion();
464 			}
465 		},
466 		/**
467 		 * 关闭某个区域的panel。
468 		 * @name omBorderLayout#closeRegion
469 		 * @function
470 		 * @param region 区域名称
471 		 * @example
472 		 * //关闭north区域的panel
473 		 * $('#page').omBorderLayout('closeRegion', 'north');
474 		 */
475 		closeRegion : function(region){
476 			var panel = this._getPanelOpts(region);
477 			if(!panel || !panel.closable){
478 				return;
479 			}
480 			var $region = this._getRegion(region);
481 				$body = $region.find(">.om-panel-body");
482 			if($region){
483 				var panelInstance = $.data($body[0],"omPanel");
484 				if(panelInstance.options.closed) return;
485 				
486 				$region.find(">.om-panel-body").omPanel("close");
487 				this._getRegionProxy(region).hide();
488 				this._resizeRegion();
489 			}
490 		},
491 		/**
492 		 * 打开某个区域的panel。
493 		 * @name omBorderLayout#openRegion
494 		 * @function
495 		 * @param region 区域名称
496 		 * @example
497 		 * //打开north区域的panel
498 		 * $('#page').omBorderLayout('openRegion', 'north');
499 		 */
500 		openRegion : function(region){
501 			var panel = this._getPanelOpts(region);
502 			if(!panel || !panel.closable){
503 				return;
504 			}
505 			var $region = this._getRegion(region);
506 				$body = $region.find(">.om-panel-body");
507 			if($region){
508 				var panelInstance = $.data($body[0],"omPanel");
509 				if(!panelInstance.options.closed) return;
510 				
511 				$region.find(">.om-panel-body").omPanel("open");
512 				this._getRegionProxy(region).hide();
513 				this._resizeRegion();
514 			}
515 		}
