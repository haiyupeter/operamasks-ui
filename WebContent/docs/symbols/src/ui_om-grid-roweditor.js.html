 60     	 * @function

 74     	/** <span style='color:red'>(作用于行编辑插件)。</span>删除某一行，如果该行是新添加的并未保存到后台，则进行物理删除；如果该行是原本就存在的，则只是隐藏并进行标记,当调用了saveChanges后才进行物理删除。
 75     	 * @function
 76     	 * @param index 行索引，从0开始 
 77     	 * @name omGrid#deleteRow
 78     	 * @returns jQuery对象
 79     	 * @example
 80     	 * $(".selector").omGrid("deleteRow" , 0);
 81     	 */
 82     	deleteRow : function(index){
 83     		if(!_isEditable(this)){
 84     			return ;
 85     		}
 86     		var $tr = this._getTrs().eq(index),
 87     			$next = $tr.next();
 88     		this.cancleEdit();
 89     		
 90     		var rowId = _getRowId($tr);
 91     		if($tr.attr("_insert")){
 92     			delete this._changeData["insert"][rowId];
 93     			$tr.remove();
 94     			if($next.hasClass("rowExpand-rowDetails")){//行详情
 95     				$next.remove();
 96     			}
 97     		}else{
 98     			this._changeData["delete"][rowId] = this._rowIdDataMap[rowId];
 99     			$tr.attr("_delete", "true").hide();
100     			if($next.hasClass("rowExpand-rowDetails")){//行详情
101         			$next.hide();
102         		}
103         	}

106         /** <span style='color:red'>(作用于行编辑插件)。</span>获取所有未保存的修改。如果没有指定type,返回的是所有的修改，格式为: {update:[],insert:[],delete:[]}，如果指定了参数，如
107     	 *  指定了"update"，则返回 [{},{}]
108     	 * @function
109     	 * @param type 可选值为："insert","update","delete" 
110     	 * @name omGrid#getChanges
111     	 * @returns 若指定了类型，返回[]，否则返回{update:[],insert:[],delete:[]}
112     	 * @example
113     	 * $(".selector").omGrid("getChanges" , "update");
114     	 */
115     	getChanges : function(type){
116     		var data = {"update":[] , "insert":[] , "delete":[]},
117     			reqType = type? type : "update",
118     			cData = this._changeData,
119     			i;
120     			
121     		if(reqType === "update"){
122     			var uData = cData[reqType];
123     			for(i in uData){
124     				own.call(uData , i) && data[reqType].push($.extend(true , {} , this._rowIdDataMap[i] , uData[i]));
125     			}
126     			reqType = type? type : "insert";
127     		}
128     		if(reqType === "insert"){
129     			var iData = cData[reqType];
130     			for(i in iData){
131     				own.call(iData , i) && data[reqType].push(iData[i]);
132     			}
133     			reqType = type? type : "delete";
134     		}
135     		if(reqType === "delete"){
136     			var dData = cData[reqType];
137     			for(i in dData){
138     				own.call(dData , i) && data[reqType].push(dData[i]);
139     			}
140     		}
141     		if(type){
142     			return data[type];
143     		}else{
144     			return data;
145     		}

148 	 	/** <span style='color:red'>(作用于行编辑插件)。</span>在指定位置动态插入一行。
149     	 * @function
150     	 * @param index 行索引，从0开始，或为"begin","end"分别表示在表格最前和最后插入行。
151     	 * @param rowData 插入的新行的初始值
152     	 * @name omGrid#insertRow
153     	 * @returns jQuery对象
154     	 * @example
155     	 * $(".selector").omGrid("insertRow" , "end" , {id:1 , name:"飞儿"});
156     	 */
157     	insertRow : function(index , rowData){
158     		var ops = this.options,
159     			cm = ops.colModel,
160     			$elem = this.element;
161     		
162     		if(!_isEditable(this)){
163     			return ;
164     		}
165     		if($.isPlainObject(index)){
166     			rowData = index;
167     			index = 0;
168     		}
169     		var $trs = this._getTrs(),
170     			rd = {};
171     		index = ("begin"==index || index==undefined)? 0 : ("end"==index? $trs.length : index);
172     
173     		for(var i=0,len=cm.length; i<len; i++){
174     			rd[cm[i]["name"]] = "";//默认都为空值
175         	}
176     		this._changeData["insert"][this._guid] = $.extend(true , rd , rowData);
177     		
178     		//创建新行
179     		var rowValues=this._buildRowCellValues(cm,rd,index),
180     			trContent = [],
181     			rowClasses=ops.rowClasses;
182     			isRowClassesFn= (typeof rowClasses === 'function'),
183     			rowCls = isRowClassesFn? rowClasses(index,rd):rowClasses[index % rowClasses.length],
184     			tdTmp = "<td align='$' abbr='$' class='grid-cell-dirty $'><div align='$' class='$' style='width:$px'>$</div></td>";//td模板
185     			    			
186     		trContent.push("<tr class='om-grid-row " + rowCls + "' _grid_row_id="+(this._guid++)+" _insert='true'>");
187     		this.hDiv.find("tr:first th").each(function(i){
188                 var axis = $(this).attr('axis'),
189                 	wrap=false,
190                 	html,
191                 	cols,
192                 	j;
193                 if(axis == 'indexCol'){
194                     html="<a class='om-icon'>新行</a>";
195                 }else if(axis == 'checkboxCol'){
196                     html = '<span class="checkbox"/>';
197                 }else if(axis.substring(0,3)=='col'){
198                     var colIndex=axis.substring(3);
199                     html=rowValues[colIndex];
200                     if(cm[colIndex].wrap){
201 						wrap=true;
202 					} 
203                 }else{
204                     html='';
205                 }
206                 cols = [this.align , this.abbr , axis , this.align , wrap?'wrap':'', $('div',$(this)).width() , html];
207                 j=0;
208                 trContent.push(tdTmp.replace(/\$/g , function(){
209                 	return cols[j++];
210                 }));
211             });
212             trContent.push("</tr>");
213     		
214     		var $tr = $(trContent.join(" ")),
215     			$destTr,
216     			$next;
217     		if(index==0){
218     			$tr.prependTo($elem.find(">tbody"));
219     		}else{
220     			$destTr = $trs.eq(index-1);
221     			$next = $destTr.next();
222     			$destTr = $next.hasClass("rowExpand-rowDetails")? $next : $destTr;//处理行详情
223     			$destTr.after($tr);
224     		}
225     		this.editRow(index);
226     	},

228     	/** <span style='color:red'>(作用于行编辑插件)。</span>保存客户端数据。注意，此方法不会提交请求到后台，而是会认为所有的数据改变都已经成功提交到后台去了，所以它会清除所有脏数据标志。
229     	 * 一般情况下，在您自己的保存方法事件回调中，调用getChanges方法获取当前所有的改变，并自己提交到后台，然后在成功回调方法中再调用本方法。
230     	 * 一旦调用本方法后，cancleChanges方法是无法对调用此方法前的所有改变起作用的。
231     	 * @function
232     	 * @name omGrid#saveChanges
233     	 * @returns jQuery对象
234     	 * @example
235     	 * $(".selector").omGrid("saveChanges");
236     	 */
237     	saveChanges : function(){
238 			if(_noChanges(this)){
239 				return ;
240 			}
241 			var uData = this._changeData["update"],
242 				$trs = this.element.find("tr.om-grid-row"),
243 				newRowsData = [];
244 			for(var i in uData){
245 				if(own.call(uData , i)){
246 					$.extend(true , this._rowIdDataMap[i] , uData[i]);
247 				}
248 			}
249 			var self = this;
250 			$trs.each(function(index , tr){
251 				var $tr = $(tr);
252 				if($tr.attr("_delete")){
253 					$tr.remove();
254 				}else{
255 					newRowsData.push(_getRowData(self , tr));
256 				}
257 			});
258 			this.pageData.data.rows = newRowsData;
259 			_clearCache(this);
260 			_resetForm(this);//重置编辑表单，清除错误信息
261 			this.refresh();//重用dataSource中的数据进行刷新
262     	},
263     	/**
264     	 * (覆盖)获取最新的所有数据
265     	 */
266     	getData : function(){
267     		var result = this.pageData.data,
268 				$trs = this._getTrs(),
269 				self = this;
270 				
271     		if(_isEditable(this) && _hasChange(this)){
272     			result = {total:result.total};
273     			result.rows = [];
274 				$trs.each(function(index , tr){
275     				result.rows.push(self._getRowData(index));
276 				});
277 			}
278 			return result;
279     	},
280     	/**
281     	 * (覆盖)获取最新的行数据。由于可以插入新的数据行，所以此方法要进行重写。
282     	 */
283     	_getRowData : function(index){
284     		var $tr = this._getTrs().eq(index),
285     			rowId = _getRowId($tr),
286     			rowData;
287     		if(_noChanges(this)){
288     			return this.pageData.data.rows[index];
289     		}
290     		if($tr.attr("_insert")){
291     			rowData = this._changeData.insert[rowId];
292     		}else{
293     			var origRowData = this._rowIdDataMap[rowId],
294     				uData = this._changeData["update"];
295     			rowData = origRowData;
296     			if(uData[rowId]){
297     				rowData = $.extend(true , {} , origRowData , uData[rowId]);
298     			}
299     		}
300     		return rowData;
301     	}

307     $.omWidget.addInitListener('om.omGrid',function(){
308     	var self = this,
309     		$elem = self.element,
310     		ops = self.options,
311     		cm = ops.colModel;
312     		
313     	self._triggered = false;//是否已经触发过一次编辑了。
314     	//如果所有列都不可编辑，那么_globalEditable=false,这时候所有原生的行都不可编辑，但如果此时设置了options.editMode=="insert",
315     	//则新添加进来的行还是可以编辑的。
316     	self._globalEditable = false;
317     	
318     	/**
319     	 * <span style='color:red'>(作用于行编辑插件)</span>编辑模式。
320     	 * 可选值为"all","insert"，默认为"all"。"all"表示所有行都是可编辑的，但如果所有的列都是不可编辑的，那么行编辑插件仍然会失效。
321     	 * 如果有这样的需求，所有的行都是不可编辑的，但是此表格可以动态添加新行，而且这些新行在持久化到后台之前是可以编辑的，那么就要使用"insert"模式了。 
322     	 * @name omGrid#editMode
323     	 * @type String
324     	 * @default "all"
325     	 * @example
326     	 * $('.selector').omGrid({width : 600, editMode:"insert");
327     	 */
328     	ops.editMode = ops.editMode || "all";//默认为all模式，即所有行都可以编辑
329     	self._allEditMode = ops.editMode=="all";
330     		
331     	for(var i=0,len=cm.length; i<len; i++){
332     		self._globalEditable = cm[i]["editor"]? true : false;
333     		if(self._globalEditable){
334     			break;
335     		}
336     	}
337     	
338     	if(!_isEditable(self)){
339     		return ;//不开启编辑功能，该插件任何方法都会失效。
340     	}
341 		self._editComps = {};//缓存每一列的编辑组件，key为colModel中的name,value为{model:模型,instance:组件实例,type:类型}
342 		self._errorHolders = {};//  name:div   每个编辑单元格对应的出错信息容器
343 		self._colWidthResize;//如果grid可编辑，在拖动标题栏改变列宽时，当前并不处于编辑状态，则设置此值为true,这样在显示编辑条时由此依据要重新计算各个组件的宽度
344 		
345 		ops._onRefreshCallbacks.push(_onRefresh);//只有数据刷新后才可以进行初始化
346 		ops._omResizableCallbacks.push(_onResizable);
347 		
348         this.tbody.delegate('tr.om-grid-row','dblclick',function(event){
349         	editRenderer.call(this , self);
350         }); 
351         this.tbody.delegate('tr.om-grid-row','click',function(event){
352         	if(self._triggered && self._editView.editing){
353 				if(self._validator){
354 					self._validator.valid() && editRenderer.call(this , self);
355 				}else{
356 					editRenderer.call(this , self);
357 				}
358         	}
359         });
360         
361         var btnScrollTimer;
362         $elem.parent().scroll(function(){
363         	if(self._triggered){
364         		if(btnScrollTimer){
365         			clearTimeout(btnScrollTimer);
366         		}
367         		btnScrollTimer = setTimeout(function(){
368     				var pos = _getEditBtnPosition(self);
369             		self._editView.editBtn.animate({"left":pos.left,"top":pos.top},self._editView.editing?"fast":0);
370             		btnScrollTimer = null;
371     			} , 300);
372         	}
373         });      
374     });

376     function editRenderer(self){
377     	var $tr = $(this),
378     		$elem = self.element,
379     		$editRow,
380     		$editForm,
381     		scrollLeft,
382     		cm = self.options.colModel,
383     		editComp,//{name:{type:组件类型,model:列对应model,instance:组件实例}}
384     		lastValue;
385     	
386     	if(!_isEditable(self)){
387     		return ;
388     	}
389     		
390     	//如果是insertEditMode模式，那么除非该行是新增的，否则不可编辑，直接返回
391     	if(!self._allEditMode && !$tr.attr("_insert")){
392     		return ;
393     	}
394     	
395     	//当前行正处于编辑状态，直接返回
396     	if(self._triggered  
397     		&& self._editView.editing
398     		&& _getRowId($tr) == self._editView.rowId){
399 			return ;        		
400     	}
401     	
402     	_showEditView(self , $tr);
403     	$editRow = self._editView.editRow;
404     	$editForm = $editRow.find(">.grid-edit-form");
405     	scrollLeft = $elem.parent().scrollLeft();
406     	
407     	self.hDiv.find("tr:first th").each(function(index){
408         	var axis = $(this).attr('axis'),
409 				model,
410 				$cell = $tr.find("td:eq("+index+")"),
411 				name,//编辑组件input域的名字，这是校验所必需的
412 				compKey;//指editComps的key
413         	if(axis.substring(0,3)=='col'){
414              	var colIndex=axis.substring(3);
415              	model = cm[colIndex];
416         	}else{
417         		if($.isEmptyObject(self._editComps)){//保证再次编辑其它行时不会重复添加 padding-left.
418         			$editRow.css("padding-left", parseInt($editRow.css("padding-left")) + $cell.outerWidth());
419         		}
420         		return ;
421         	}
422         	var editor = model.editor;
423         	if(!self._triggered){
424         		//如果列不可编辑，则默认type="text"
425             	//列不可编辑的条件: 
426             	//(1)colModel没有editor属性
427             	//(2)colModel有editor属性，并且editor有editable属性，那么editable===false则不进行编辑，或者editable为函数且返回false也不进行编辑。
428 				if(!editor || 
429 	            	(editor && 
430 	            		(editor.editable===false || 
431 	                    	($.isFunction(editor.editable) && editor.editable()===false) ) ) ){
432 					var renderer = editor && editor.renderer;
433 	       			model.editor = editor = {};
434 	       			editor.type = "text";
435 	       			if(renderer){
436 	       				editor.renderer = renderer;
437 	       				editor.type = "custom";
438 	       			}
439 	       			editor.editable = false;
440 	   			}else{
441 	   				editor.type = editor.type || "text";
442 	   				editor.editable = true;
443 	   				if(editor.rules){
444 	   					self._validate = true;//只有需要检验才需要进行检验
445 	   				}
446 	   			}
447 	   			compKey = model.editor.name || model.name;
448 	   			editor.options = editor.options || {};
449 	   			self._editComps[compKey] = {};
450         	}else{
451         		compKey = model.editor.name || model.name;
452         	}
453         	editComp = self._editComps[compKey];
454    			lastValue = _getLastValue(self , $tr , model) || "";
455    			
456    			//可编辑并且可校验，添加对应的出错信息显示容器
457    			if(!self._triggered && editor.editable && editor.rules){
458    				self._errorHolders[compKey] = $("<div class='errorContainer' style='display:none'></div>").appendTo($elem.parent());
459    			}
460    			var $ins = editComp.instance,
461    				$wrapper,
462    				type = editor.type;
463    			if(!$ins){
464    				$wrapper = $("<div style='position:absolute'></div>").css({left:$cell.position().left+scrollLeft,top:3}).appendTo($editForm).addClass("grid-edit-wrapper"); 
465    				$ins = editComp.instance = $("<input></input>").attr({"name":compKey,"id":compKey}).appendTo($wrapper);
466    				if("text"!=type && "custom"!=type){//实例化组件
467    					$ins[type](editor.options);
468    				}
469    				if("omCalendar"==type || "omCombo"==type){
470    					var $parent = $ins.parent();
471    					if("omCalendar"==type){
472    						$ins.val(lastValue).width($cell.outerWidth()-24);
473    					}
474    					$ins.width($cell.outerWidth(true) - ($parent.outerWidth(true) - $ins.width()));
475    				}else{
476    					if("text"==type){
477    						$ins.addClass("grid-edit-text");
478    						if(!editor.editable){
479    							$ins.attr("readonly" , "readonly").addClass("readonly-text");
480    						}
481    					}
482    					if("custom"==type){
483    						$wrapper.html(editor.renderer(lastValue));
484    					}
485    					$ins.width($cell.outerWidth(true) - ($ins.outerWidth(true) - $ins.width()));
486    				}
487    				editComp.model = model;
488 				editComp.type = type;
489 				editComp.id = model.name;
490 				if("custom" != type){//非"custom"类型失去焦点要隐藏错误信息
491 					$ins.blur(function(event){
492 						var eHolder = self._errorHolders[compKey];
493 						eHolder && eHolder.hide();	
494 					});
495 				}
496    			}
497 			switch(type){
498 				case "omCalendar":
499 					$ins = $ins.val(lastValue).omCalendar();
500 					break;
501 				case "omNumberField":
502 					$ins.val(lastValue).trigger("blur");//进行错误处理
503 					break;
504 				case "omCombo":
505 					$ins.omCombo("value" , lastValue);
506 					break;
507 				case "text":
508 					$ins.val(lastValue);
509 					break;
510 				default:;
511 			}
512         });

713 	function _saveEditValue(self , tr){
714 		var $tr = $(tr),
715 			$editRow = self._editView.editRow,
716 			comps = self._editComps,
717 			rowId = _getRowId($tr),
718 			index = self._getTrs().index($tr),
719 			rowData = self._getRowData(index);
720 			
721 		$.each(comps , function(name , comp){
722 			var key = comp.model.name,
723 				newValue = _getCompValue(self , $tr , comp),
724 				originalValue,
725 				html,
726 				updateRowData;
727 				
728 			if($tr.attr("_insert")){
729 				self._changeData.insert[rowId][key] = newValue;
730 			}else{
731 				originalValue = _getRowData(self , tr)[key];
732 				updateRowData = self._changeData.update[rowId];
733 				if(newValue == originalValue){
734 					_toggleDirtyFlag($tr , comp.model , false);
735 					updateRowData && delete updateRowData[key];
736 					$.isEmptyObject(updateRowData) && delete self._changeData.update[rowId];
737 				}else{
738 					_toggleDirtyFlag($tr , comp.model , true);
739 					updateRowData = self._changeData.update[rowId] = updateRowData || {};
740 					updateRowData[key] = newValue;
741 				}
742 			}
743 			//更新回表格
744 			if(comp.model.renderer){
745 				html = comp.model.renderer(newValue , rowData ,index);
746 			}else{
747 				html = newValue? newValue : "";
748 			}
749 			$tr.find("td[abbr='"+key+"'] >div").html(html);
750 		});
751 	}

797 			}

818 					msg = messages[model.editor.name || model.name] = {};
819 				if(customRules.length>0 && !$.isArray(customRules[0])){
820 					var temp = [];
821 					temp.push(customRules);//包装成[[],[]]这种统一形式

823 				}
824 				for(var i=0,len=customRules.length; i<len; i++){
825 					var name = customRules[i][0];//检验类型
826 					r[name]  = customRules[i][1] == undefined? true : customRules[i][1]; //没有定义值的统一传 true
827 					if(customRules[i][2]){
828 						msg[name] = customRules[i][2];
829 					}
