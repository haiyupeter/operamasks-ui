 15 (function($) {
 16 	/**
 17 	 * 分页导航按钮样式与类别
 18 	 */
 19 	var pagerNvg = [
 20 		{cls:"pFirst" , type:"first"},
 21 		{cls:"pPrev" , type:"prev"},
 22 		{cls:"pNext" , type:"next"},
 23 		{cls:"pLast" , type:"last"},
 24 		{cls:"pReload" , type:"reload"}
 25 	];	
 26 	function needChange(self , type){
 27 		var oldPage = self._oldPage,
 28 		    nowPage = self.pageData.nowPage;
 29 		if("input" === type){
 30 			return oldPage != $('.pControl input', self.element.closest('.om-grid')).val();
 31 		}else if("reload" === type){
 32 			return true;
 33 		}else{
 34 			return oldPage != nowPage;
 35 		}
 36 	}
 37 	
 38     $.omWidget.addInitListener('om.omGrid',function(){
 39         var self = this,
 40             cm = this._getColModel(),
 41             tds = this._getHeaderCols().filter("[axis^='col']"),
 42             $pDiv = this.pDiv;
 43         
 44         $(tds).each(function(index){
 45             var sortFn=cm[index].sort;
 46             if(sortFn){
 47                 var _this=$(this).click(function(){
 48                     var sortCol = cm[index].name;
 49                     var removeClass = _this.hasClass('asc')?'asc'
 50                                     : _this.hasClass('desc')?'desc'
 51                                     : null;
 52                     var sortDir=(removeClass=='asc'?'desc':'asc');
 53                     tds.removeClass('asc desc');
 54                     _this.addClass(sortDir);
 55                     switch(sortFn){
 56                         case 'serverSide':
 57                             var extraData=self.options.extraData;
 58                             extraData.sortBy=sortCol;
 59                             extraData.sortDir=sortDir;
 60                             self.reload();
 61                             return;
 62                         case 'clientSide':
 63                             sortFn=function(obj1,obj2){
 64                                 var v1=obj1[sortCol],v2=obj2[sortCol];
 65                                 return v1==v2?0:v1>v2?1:-1;
 66                             };
 67                             break;
 68                         default:
 69                             // do nothing,keep sortFn==cm[index].sort
 70                     }
 71                     var datas = self.pageData.data;
 72                     if(removeClass==null){//从未排序变成升序排列
 73                         datas.rows=datas.rows.sort(sortFn);
 74                     }else{//升序变成降序，或降序变成升序，只需要反转数据即可
 75                         datas.rows=datas.rows.reverse();
 76                     }
 77                     self.refresh();
 78                 });
 79                 _this.children().first().append('<img class="om-grid-sortIcon" src="data:image/gif;base64,R0lGODlhAQABAID/AMDAwAAAACH5BAEAAAAALAAAAAABAAEAAAICRAEAOw=="></img>');
 80             }
 81         });
 82         //监听分页条按钮的事件，如果为客户端排序，每次都要清除掉排序的状态
 83         for(var i=0,len=pagerNvg.length; i<len; i++){
 84         	(function(i){
 85         		$pDiv.find("."+pagerNvg[i].cls).click(function(){
 86 					var change = needChange(self , pagerNvg[i].type);
 87 					tds.each(function(index){
 88 						var $headerCol = $(this);
 89 						if(change && ($headerCol.hasClass('asc') || $headerCol.hasClass('desc')) &&  "serverSide" !== cm[index].sort){
 90 							$headerCol.removeClass('asc desc');
 91 						}
 92 					});
 93 				});	 
 94         	})(i);
 95 		}
 96         $('.pControl input', $pDiv).keydown(function(e){
 97         	var change = needChange(self , "input");
 98         	if (e.keyCode == $.om.keyCode.ENTER) {
 99         		tds.each(function(index){
100 					var $headerCol = $(this);
101 					if(change && ($headerCol.hasClass('asc') || $headerCol.hasClass('desc')) &&  "serverSide" !== cm[index].sort){
102 						$headerCol.removeClass('asc desc');
103 					}
104 				});
105         	}
106         });
107         
108         /**
109 	     * 清空omGrid的排序状态。比如先在第一列上进行了降序排列，以后每次取数时都是按这个降序排列条件来取数，如果要清空排序条件，调用本方法即可，调用后再次取数时就跟从未进行过排序一样。<br/>
110 	     * <b>注意：此方法仅清空排序状态，并不立即取数。</b>
111 	     * @function
112 	     * @example
113 	     *  $('.selector').omGrid('clearSort');
114 	     */
115 	    this.clearSort=function(){
116 	        var extraData = this.options.extraData;
117 	        extraData.sortBy = undefined;
118 	        extraData.sortDir = undefined;
119 	        this._getHeaderCols().removeClass('asc desc');
120 	        //$('tr:first th[axis^="col"]',this.thead).removeClass('asc desc');
121 	    };
122     });
123 })(jQuery);
