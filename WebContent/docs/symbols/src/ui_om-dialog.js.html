406 	/**
407      * 关闭对话框.
408      * @name omDialog#close
409      * @function
410      * @returns 支持链式操作，返回JQuery对象
411      * @example
412      * var store = $("#select").omDialog('close');
413      * 
414      */
415 	close: function(event) {
416 		var self = this,
417 			maxZ, thisZ,
418 			options = this.options,
419 			onBeforeClose = options.onBeforeClose,
420 			onClose = options.onClose;
421 		
422 		if (onBeforeClose && false === self._trigger("onBeforeClose",event)) {
423 			return;
424 		}
425 
426 		if (self.overlay) {
427 			self.overlay.destroy();
428 		}
429 		self.uiDialog.unbind('keypress.om-dialog');
430 
431 		self._isOpen = false;
432 
433 		if (self.options.hide) {
434 			self.uiDialog.hide(self.options.hide, function() {
435                 onClose && self._trigger("onClose",event);
436 			});
437 		} else {
438 			self.uiDialog.hide();
439 			onClose && self._trigger("onClose",event);
440 		}
441 
442 		$.om.omDialog.overlay.resize();
443 
444 		// adjust the maxZ to allow other modal dialogs to continue to work (see #4309)
445 		if (self.options.modal) {
446 			maxZ = 0;
447 			$('.om-dialog').each(function() {
448 				if (this !== self.uiDialog[0]) {
449 					thisZ = $(this).css('z-index');
450 					if(!isNaN(thisZ)) {
451 						maxZ = Math.max(maxZ, thisZ);
452 					}
453 				}
454 			});
455 			$.om.omDialog.maxZ = maxZ;
456 		}
457 
458 		return self;
459 	},

474 	// the force parameter allows us to move modal dialogs to their correct
475 	// position on open
476 	moveToTop: function(force, event) {
477 		var self = this,
478 			options = self.options,
479 			saveScroll;
480 
481 		if ((options.modal && !force) ||
482 			(!options.stack && !options.modal)) {
483 			return self._trigger('onFocus', event);
484 		}
485 
486 		if (options.zIndex > $.om.omDialog.maxZ) {
487 			$.om.omDialog.maxZ = options.zIndex;
488 		}
489 		if (self.overlay) {
490 			$.om.omDialog.maxZ += 1;
491 			self.overlay.$el.css('z-index', $.om.omDialog.overlay.maxZ = $.om.omDialog.maxZ);
492 		}
493 
494 		//Save and then restore scroll since Opera 9.5+ resets when parent z-Index is changed.
495 		//  http://ui.jquery.com/bugs/ticket/3193
496 		saveScroll = { scrollTop: self.element.scrollTop(), scrollLeft: self.element.scrollLeft() };
497 		$.om.omDialog.maxZ += 1;
498 		self.uiDialog.css('z-index', $.om.omDialog.maxZ);
499 		self.element.attr(saveScroll);
500 		self._trigger('onFocus', event);
501 
502 		return self;
503 	},

505 	/**
506      * 打开对话框。
507      * @name omDialog#open
508      * @function
509      * @returns 支持链式操作，返回JQuery对象
510      * @example
511      * var store = $("#select").omDialog('open');
512      * 
513      */
514 	open: function() {
515 		if (this._isOpen) { return; }
516 
517 		var self = this,
518 			options = self.options,
519 			uiDialog = self.uiDialog;
520 
521 		self.overlay = options.modal ? new $.om.omDialog.overlay(self) : null;
522 		self._size();
523 		self._position(options.position);
524 		uiDialog.show(options.show);
525 		self.moveToTop(true);
526 
527 		// prevent tabbing out of modal dialogs
528 		if (options.modal) {
529 			uiDialog.bind('keypress.om-dialog', function(event) {
530 				if (event.keyCode !== $.om.keyCode.TAB) {
531 					return;
532 				}
533 
534 				var tabbables = $(':tabbable', this),
535 					first = tabbables.filter(':first'),
536 					last  = tabbables.filter(':last');
537 
538 				if (event.target === last[0] && !event.shiftKey) {
539 					first.focus(1);
540 					return false;
541 				} else if (event.target === first[0] && event.shiftKey) {
542 					last.focus(1);
543 					return false;
544 				}
545 			});
546 		}
547 
548 		// set focus to the first tabbable element in the content area or the first button
549 		// if there are no tabbable elements, set focus on the dialog itself
550 		$(self.element.find(':tabbable').get().concat(
551 			uiDialog.find('.om-dialog-buttonpane :tabbable').get().concat(
552 				uiDialog.get()))).eq(0).focus();
553 
554 		self._isOpen = true;
555 		var onOpen = options.onOpen;
556 		if(onOpen){
557 		    self._trigger("onOpen");
558 		}
559 		return self;
560 	},

562 	_createButtons: function(buttons) {
563 		var self = this,
564 			hasButtons = false,
565 			uiDialogButtonPane = $('<div></div>')
566 				.addClass(
567 					'om-dialog-buttonpane ' +
568 					'om-widget-content ' +
569 					'om-helper-clearfix'
570 				),
571 			uiButtonSet = $( "<div></div>" )
572 				.addClass( "om-dialog-buttonset" )
573 				.appendTo( uiDialogButtonPane );
574 
575 		// if we already have a button pane, remove it
576 		self.uiDialog.find('.om-dialog-buttonpane').remove();
577 
578 		if (typeof buttons === 'object' && buttons !== null) {
579 			$.each(buttons, function() {
580 				return !(hasButtons = true);
581 			});
582 		}
583 		if (hasButtons) {
584 			$.each(buttons, function(name, props) {
585 				props = $.isFunction( props ) ?
586 					{ click: props, text: name } :
587 					props;
588 				var button = $('<button type="button"></button>')
589 					.click(function() {
590 						props.click.apply(self.element[0], arguments);
591 					})
592 					.appendTo(uiButtonSet);
593 				// can't use .attr( props, true ) with jQuery 1.3.2.
594 				$.each( props, function( key, value ) {
595 					if ( key === "click" ) {
596 						return;
597 					}
598 					if ( key in attrFn ) {
599 						button[ key ]( value );
600 					} else {
601 						button.attr( key, value );
602 					}
603 				});
604 				if ($.fn.omButton) {
605 					button.omButton();
606 				}
607 			});
608 			uiDialogButtonPane.appendTo(self.uiDialog);
609 		}

612 	_makeDraggable: function() {
613 		var self = this,
614 			options = self.options,
615 			doc = $(document),
616 			heightBeforeDrag;
617 
618 		function filteredUi(ui) {
619 			return {
620 				position: ui.position,
621 				offset: ui.offset
622 			};
623 		}
624 
625 		self.uiDialog.omDraggable({
626 			cancel: '.om-dialog-content, .om-dialog-titlebar-close',
627 			handle: '.om-dialog-titlebar',
628 			containment: 'document',
629 			cursor: 'move',
630 			onStart: function(ui, event) {
631 				heightBeforeDrag = options.height === "auto" ? "auto" : $(this).height();
632 				$(this).height($(this).height()).addClass("om-dialog-dragging");
633 				self._trigger('onDragStart', filteredUi(ui), event);
634 			},
635 			onDrag: function(ui, event) {
636 				self._trigger('onDrag', filteredUi(ui), event);
637 			},
638 			onStop: function(ui, event) {
639 				options.position = [ui.position.left - doc.scrollLeft(),
640 					ui.position.top - doc.scrollTop()];
641 				$(this).removeClass("om-dialog-dragging").height(heightBeforeDrag);
642 				self._trigger('onDragStop', filteredUi(ui), event);
643 				$.om.omDialog.overlay.resize();
644 			}
645 		});
646 	},

648 	_makeResizable: function(handles) {
649 		handles = (handles === undefined ? this.options.resizable : handles);
650 		var self = this,
651 			options = self.options,
652 			// .ui-resizable has position: relative defined in the stylesheet
653 			// but dialogs have to use absolute or fixed positioning
654 			position = self.uiDialog.css('position'),
655 			resizeHandles = (typeof handles === 'string' ?
656 				handles	:
657 				'n,e,s,w,se,sw,ne,nw'
658 			);
659 
660 		function filteredUi(ui) {
661 			return {
662 				originalPosition: ui.originalPosition,
663 				originalSize: ui.originalSize,
664 				position: ui.position,
665 				size: ui.size
666 			};
667 		}
668 
669 		self.uiDialog.omResizable({
670 			cancel: '.om-dialog-content',
671 			containment: 'document',
672 			alsoResize: self.element,
673 			maxWidth: options.maxWidth,
674 			maxHeight: options.maxHeight,
675 			minWidth: options.minWidth,
676 			minHeight: self._minHeight(),
677 			handles: resizeHandles,
678 			start: function(event, ui) {
679 				$(this).addClass("om-dialog-resizing");
680 				self._trigger('onResizeStart', event, filteredUi(ui));
681 			},
682 			resize: function(event, ui) {
683 				self._trigger('onResize', event, filteredUi(ui));
684 			},
685 			stop: function(event, ui) {
686 				$(this).removeClass("om-dialog-resizing");
687 				options.height = $(this).height();
688 				options.width = $(this).width();
689 				self._trigger('onResizeStop', event, filteredUi(ui));
690 				$.om.omDialog.overlay.resize();
691 			}
692 		})
693 		.css('position', position)
694 		.find('.om-resizable-se').addClass('om-icon om-icon-grip-diagonal-se');
695 	},

697 	_minHeight: function() {
698 		var options = this.options;
699 
700 		if (options.height === 'auto') {
701 			return options.minHeight;
702 		} else {

710 			isVisible;

781 		var self = this,
782 			uiDialog = self.uiDialog;
783 
784 		switch (key) {
785 			case "buttons":
786 				self._createButtons(value);
787 				break;
788 			case "closeText":
789 				// ensure that we always pass a string
790 				self.uiDialogTitlebarCloseText.text("" + value);
791 				break;
792 			case "dialogClass":
793 				uiDialog
794 					.removeClass(self.options.dialogClass)
795 					.addClass(uiDialogClasses + value);
796 				break;
797 			case "disabled":
798 				if (value) {
799 					uiDialog.addClass('om-dialog-disabled');
800 				} else {
801 					uiDialog.removeClass('om-dialog-disabled');
802 				}
803 				break;
804 			case "draggable":
805 				var isDraggable = uiDialog.is( ":data(draggable)" );
806 				if ( isDraggable && !value ) {
807 					uiDialog.omDraggable( "destroy" );
808 				}
809 				
810 				if ( !isDraggable && value ) {
811 					self._makeDraggable();
812 				}
813 				break;
814 			case "position":
815 				self._position(value);
816 				break;
817 			case "resizable":
818 				// currently resizable, becoming non-resizable
819 				var isResizable = uiDialog.is( ":data(resizable)" );
820 				if (isResizable && !value) {
821 					uiDialog.omResizable('destroy');
822 				}
823 
824 				// currently resizable, changing handles
825 				if (isResizable && typeof value === 'string') {
826 					uiDialog.omResizable('option', 'handles', value);
827 				}
828 
829 				// currently non-resizable, becoming resizable
830 				if (!isResizable && value !== false) {
831 					self._makeResizable(value);
832 				}
833 				break;
834 			case "title":
835 				// convert whatever was passed in o a string, for html() to not throw up
836 				$(".om-dialog-title", self.uiDialogTitlebar).html("" + (value || ' '));
837 				break;
838 		}
839

843 	_size: function() {
844 		/* If the user has resized the dialog, the .ui-dialog and .ui-dialog-content
845 		 * divs will both have width and height set, so we need to reset them
846 		 */
847 		var options = this.options,
848 			nonContentHeight,
849 			minContentHeight,
850 			isVisible = this.uiDialog.is( ":visible" );
851 
852 		// reset content sizing
853 		this.element.show().css({
854 			width: 'auto',
855 			minHeight: 0,
856 			height: 0
857 		});
858 
859 		if (options.minWidth > options.width) {
860 			options.width = options.minWidth;
861 		}
862 
863 		// reset wrapper sizing
864 		// determine the height of all the non-content elements
865 		nonContentHeight = this.uiDialog.css({
866 				height: 'auto',
867 				width: options.width
868 			})
869 			.height();
870 		minContentHeight = Math.max( 0, options.minHeight - nonContentHeight );
871 		
872 		if ( options.height === "auto" ) {
873 			// only needed for IE6 support
874 			if ( $.support.minHeight ) {
875 				this.element.css({
876 					minHeight: minContentHeight,
877 					height: "auto"
878 				});
879 			} else {
880 				this.uiDialog.show();
881 				var autoHeight = this.element.css( "height", "auto" ).height();
882 				if ( !isVisible ) {
883 					this.uiDialog.hide();
884 				}
885 				this.element.height( Math.max( autoHeight, minContentHeight ) );
886 			}
887 		} else {
888 			this.element.height( Math.max( options.height - nonContentHeight, 0 ) );
889 		}
890 
891 		if (this.uiDialog.is(':data(resizable)')) {
892 			this.uiDialog.omResizable('option', 'minHeight', this._minHeight());
893 		}
894 	}
