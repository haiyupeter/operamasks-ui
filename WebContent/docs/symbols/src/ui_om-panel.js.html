308 		_create: function(){
309 		    this.element.addClass("om-panel-body om-widget-content")
310 		    	.wrap("<div class='om-widget om-panel'></div>");
311 		},
312 		_init: function(){
313 			var options = this.options,
314 				$body = this.element,
315 				$parent = $body.parent(),
316 				$header;
317 			this._renderHeader();
318 			$header = $body.prev();
319 			if(options.header === false){
320 		 		$body.addClass("om-panel-noheader");
321 		 	}
322 			this._bindEvent();
323 		 	this._resize($parent);
324 		 	var headerHeight = options.header !== false? $header.outerHeight() : 0;
325 		 	if(options.collapsed !== false){
326 		 		"auto"!==options.height && $parent.height(headerHeight);		 		
327 		 		$body.hide();
328 		 		if(options.header !== false){
329 		 			$header.find(">.om-panel-tool >.om-panel-tool-collapse").removeClass("om-panel-tool-collapse")
330 		 				.addClass("om-panel-tool-expand");
331 		 		}
332 		 	}else{
333 		 		$body.show();
334 		 		"auto"!==options.height && $parent.height(headerHeight + $body.outerHeight());
335 		 		if(options.header !== false){
336 		 			$header.find(">.om-panel-tool >.om-panel-tool-expand").removeClass("om-panel-tool-expand")
337 		 				.addClass("om-panel-tool-collapse");
338 		 		}	
339 		 	}
340 		 	options.closed !== false? this._hide($parent) : this._show($parent);
341 		 	this.reload();
342 		},
343 		_hide: function($target){
344 			if("hidden" === this.options._closeMode){
345 				$target.hide();
346 			}else if("visibility" === this.options._closeMode){
347 				$target.addClass("om-helper-hidden-accessible");
348 			}
349 		},
350 		_show: function($target){
351 			if("hidden" === this.options._closeMode){
352 				$target.show();
353 			}else if("visibility" === this.options._closeMode){
354 				$target.removeClass("om-helper-hidden-accessible");
355 			}
356 		},
357 		_bindEvent: function(){
358 			var self = this,
359 				$body = this.element,
360 				options = this.options,
361 				tools = $body.prev().find("> .om-panel-tool");
362 				
363 			if(options.collapsible){
364 				tools.find(">.om-panel-tool-collapse , .om-panel-tool-expand")
365 					.click(function(){
366 						if(options.collapsed !== false){
367 							self.expand();
368 						}else{
369 							self.collapse();
370 						}
371 					});
372 			}
373 			if(options.closable !== false){
374 				tools.find(">.om-panel-tool-close")
375 					.click(function(){
376 						self.close();
377 					});
378 			}
379 			$(window).bind("resize.ompanel" , function(){
380 				$frame = $body.find(">.panel-inner-frame");
381 				if($frame.length !== 0 && options.width==='auto'){
382 					$frame.width($body.innerWidth());
383 				}
384 			});
385 		},
386 		_renderHeader: function(){
387 			this.header && this.header.remove();
388 			if(this.options.header === false){
389 				return ;
390 			}
391 			var that = this,
392 				options = this.options,
393 				tools = options.tools,
394 				$header = this.header = $("<div class='om-panel-header'></div>").insertBefore(this.element);
395 			if(options.iconCls){
396 				$("<div class='om-icon om-panel-icon'></div>").addClass(options.iconCls).appendTo($header);
397 			}
398 			$("<div class='om-panel-title'></div>").html(options.title).appendTo($header);
399 			$tool = $("<div class='om-panel-tool'></div>");
400 			if(options.collapsible !== false){
401 				$("<div class='om-icon om-panel-tool-collapse'></div>").appendTo($tool);	
402 			}
403 			//处理自定义头部右边的工具条
404 			if($.isArray(tools)){
405 				for(var i=0,len=tools.length; i<len; i++){
406 					var tool = tools[i],
407 						iconCls;
408 					if(iconCls = this._getInnerToolCls(tool.id)){
409 						$("<div class='om-icon'></div>").addClass(iconCls)
410 							.click(	function(event){
411 								tool.handler.call(this,that,event);
412 							}).appendTo($tool);
413 					}else if(typeof tool.iconCls === 'string'){
414 						$("<div class='om-icon'></div>").addClass(tool.iconCls)
415 							.click(	function(event){
416 								tool.handler.call(this,that,event);
417 							}).appendTo($tool);
418 					}else if($.isArray(tool.iconCls)){
419 						//这里必须要用内部匿名函数，因为hover中用到了tool，否则tool的值很可能已经被改掉了
420 						(function(tool){
421 							$("<div class='om-icon'></div>").addClass(tool.iconCls[0])
422 								.click(function(event){
423 									tool.handler.call(this,that,event);
424 								})
425 								.hover(function(){
426 									if(tool.iconCls[1]){
427 										$(this).toggleClass(tool.iconCls[1]);
428 									}
429 								}).appendTo($tool);
430 						})(tool);
431 					}
432 				}
433 			}else{
434 				try{
435 					$(tools).appendTo($tool);
436 				}catch(error){
437 					throw "bad format of jquery selector.";
438 				}
439 			}
440 			
441 			if(options.closable !== false){
442 				$("<div class='om-icon om-panel-tool-close'></div>").appendTo($tool);	
443 			}
444 			//处理内置工具按钮hover时的样式变换
445 			$tool.find(">div.om-icon").hover(
446 				function(){
447 					var self = this;
448 					$.each(innerToolCls , function(){
449 						if($(self).hasClass(this)){
450 							$(self).toggleClass(this+"-hover");
451 						}
452 					});
453 				}
454 			);
455 			$tool.appendTo($header);
456 		},
457 		/**
458 		 * 初始化panel,header,body的宽和高
459 		 */
460 	 	_resize: function($panel){
461 	 		var $body = this.element,
462 	 			$header = $body.prev(),
463 	 			$panel = $body.parent(),
464 	 			options = this.options;
465 	 		if(options.width == 'fit'){
466 	 			options.width = '100%';
467 	 			$panel.width('100%');
468 	 			$header.css("width" , "");
469 	 			$body.css("width" , "");
470 	 		}else if(options.width !== 'auto'){
471 				$panel.width(options.width);
472 				$header.outerWidth($panel.width());
473 				$body.outerWidth($panel.width());
474 	 		}else{
475 	 			var style = $body.attr("style");
476 	 			if(style && style.indexOf("width") !== -1){
477 	 				$panel.width($body.outerWidth());
478 	 				$header.outerWidth($body.outerWidth());
479 	 			}else{
480 	 				$panel.css("width" , "");
481 		 			$header.css("width" , "");
482 		 			$body.css("width" , "");
483 	 			}
484 	 		}
485 	 		if(options.height == 'fit'){
486 	 			options.height = '100%';
487 	 			$panel.height('100%');
488 	 			$body.outerHeight($panel.height()- (this.options.header!==false?$header.outerHeight():0) );	 
489 	 		}else if(options.height !== 'auto'){
490 				$panel.height(options.height);
491 				$body.outerHeight($panel.height()- (this.options.header!==false?$header.outerHeight():0) );	 
492 	 		}else{
493 	 			var style = $body.attr("style");
494 	 			if(style && style.indexOf("height") !== -1){
495 	 				$panel.height($header.outerHeight() + $body.outerHeight());
496 	 			}else{
497 	 				$panel.css("height" , "");
498 		 			$body.css("height" , "");
499 	 			}
500 	 		}
501 	 	},
502 	 	_getInnerToolCls: function(id){
503 	 		return $.inArray(id , innerToolId)!=-1? 'om-panel-tool-'+id : null;
504 	 	},
505 		_showLoadingMessage: function(){
506 			var options = this.options,
507 				$body = this.element,
508 				$loadMsg = $body.next(".om-panel-loadingMessage"),
509 				position = {
