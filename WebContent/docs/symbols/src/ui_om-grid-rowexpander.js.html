 15 (function($) {
 16     $.omWidget.addInitListener('om.omGrid',function(){
 17         var self=this,
 18             rowDetailsProvider=self.options.rowDetailsProvider; //渲染行详情的function
 19         if(!rowDetailsProvider){
 20             return;
 21         }
 22         $('tr th:eq(0)',self.thead).before('<th align="center" axis="expenderCol" class="expenderCol"><div style="text-align: center; width: 14px;"></div></th>');
 23         //修正各列宽度
 24         var autoExpandColIndex = -1,
 25             cm=self.options.colModel,
 26             allColsWidth=0;
 27         for(var i=0,len=cm.length;i<len;i++){
 28             if(cm[i].width=='autoExpand'){
 29                 autoExpandColIndex = i;
 30             }else{
 31                 allColsWidth+=cm[i].width;
 32             }
 33         }
 34         if(autoExpandColIndex != -1){ //说明有某列要自动扩充
 35             var toBeExpandedTh = self.thead.find('th[axis="col'+autoExpandColIndex+'"]>div');
 36             toBeExpandedTh.width(toBeExpandedTh.width()-self.thead.find('th[axis="expenderCol"]').width());
 37         }else if(self.options.autoFit){
 38             var percent=self.thead.find('th[axis="expenderCol"]').width()/allColsWidth,
 39                 toFixedThs=tr.find('th[axis^="col"]>div');
 40             toFixedThs.each(function(index){
 41                 $(this).css('width',parseInt($(this).width()-cm[index].width*percent));
 42             });
 43         }
 44         var colCount=$('tr th',self.thead).size();//总共列数
 45         self.tbody.delegate('tr td.expenderCol>div','click',function(event){
 46             var _this=$(this),
 47                 thisRow=_this.closest('tr'),
 48                 nextTr=thisRow.next('tr');
 49             if(nextTr.hasClass('rowExpand-rowDetails')){ //已经构造过了，直接显示/隐藏
 50                 nextTr.toggle();
 51             }else{ //没有构造过，则构造并显示
 52                 var rowIndex=thisRow.index('tr.om-grid-row'),
 53                     rowData=self._getRowData(rowIndex),
 54                     rowDetails=rowDetailsProvider?rowDetailsProvider(rowData,rowIndex):' ';
 55                 thisRow.after('<tr class="rowExpand-rowDetails"><td colspan="'+(colCount-1)+'"><div class="rowExpand-rowDetails-content">'+rowDetails+'</div></td></tr>');
 56             }
 57             if(_this.hasClass('rowExpand-expanded')){
 58                 _this.removeClass('rowExpand-expanded').parent().attr('rowspan',1);
 59             }else{
 60                 _this.addClass('rowExpand-expanded').parent().attr('rowspan',2);
 61             }
 62             return false; //不触发onRowSelect和onRowClick事件
 63         });
 64     });
 65 })(jQuery);
